def _build_topology(self, mined: ProductionBlueprint) -> Tuple[Dict, List]:
        """根据必需节点列表和常见连接构建拓扑"""
        required = list(mined.required_nodes)
        if not required:
            return {}, []

        # 分阶段排序
        ordered = self._order_pipeline(required, mined.category)
        if not ordered:
            return {}, []

        workflow = {}
        edges = []
        node_refs = {}  # class_type → node_id (可能有多个同名节点)

        # 特殊处理：CLIPTextEncode 需要两个（positive + negative）
        multi_instances = defaultdict(int)
        for ct in ordered:
            multi_instances[ct] += 1

        # 处理那些出现多次的节点
        clip_encode_count = multi_instances.get("CLIPTextEncode", 0)
        clip_encode_placed = 0

        for ct in ordered:
            nid = self._next_id()

            # 获取节点定义
            node_def = KNOWN_SLOT_MAP.get(ct, {})
            if not node_def:
                # 未知节点：创建空壳
                workflow[nid] = {"class_type": ct, "inputs": {}}
                node_refs[ct] = nid
                continue

            inputs = {}
            for input_name, (source_ct, source_slot) in node_def.items():
                if not source_ct:
                    # widget 输入：使用默认参数
                    inputs[input_name] = self._default_param(ct, input_name, mined.category)
                else:
                    # 连接输入
                    src_nid = node_refs.get(source_ct)
                    if src_nid is not None:
                        inputs[input_name] = [src_nid, source_slot]

            workflow[nid] = {"class_type": ct, "inputs": inputs}

            # 处理多实例
            if ct == "CLIPTextEncode":
                if clip_encode_placed == 0:
                    node_refs["CLIPTextEncode"] = nid
                clip_encode_placed += 1
            else:
                node_refs[ct] = nid

        # 后处理：确保连接完整性
        workflow = self._wire_connections(workflow, ordered, mined.common_edges)

        return workflow, edges
