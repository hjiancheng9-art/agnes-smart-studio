# flow_compiler.py
from __future__ import annotations

import json
import re
from dataclasses import dataclass
from typing import Any, Callable

try:
    from jsonschema import Draft202012Validator
except Exception:
    Draft202012Validator = None


COMFY_PROMPT_SCHEMA = {
    "type": "object",
    "minProperties": 1,
    "patternProperties": {
        r"^\d+$": {
            "type": "object",
            "required": ["class_type", "inputs"],
            "additionalProperties": True,
            "properties": {
                "class_type": {"type": "string", "minLength": 1},
                "inputs": {"type": "object"},
                "_meta": {"type": "object"},
            },
        }
    },
    "additionalProperties": False,
}


DEFAULT_SINK_CLASSES = {
    "SaveImage",
    "PreviewImage",
    "SaveAnimatedWEBP",
    "VHS_VideoCombine",
    "SaveVideo",
    "VideoCombine",
}


class FlowValidationError(Exception):
    def __init__(self, errors: list[str]):
        super().__init__("\n".join(errors))
        self.errors = errors


@dataclass
class FlowValidator:
    """
    ComfyUI Prompt JSON validator.

    校验：
    1. JSON Schema 基础结构
    2. class_type 是否来自 /object_info，防幻觉 node
    3. input key 是否属于 object_info 定义
    4. 引用是否合法：[node_id, output_index]
    5. DAG 无环
    6. 所有节点都能追溯到输出节点，防拓扑不连通 / 孤儿节点
    """

    node_catalog: dict[str, Any]
    sink_classes: set[str] | None = None

    def __post_init__(self):
        self.allowed_classes = set(self.node_catalog.keys())
        self.sink_classes = self.sink_classes or DEFAULT_SINK_CLASSES

    def parse_and_validate(self, text: str) -> dict[str, Any]:
        prompt = self.extract_json(text)
        self.validate(prompt)
        return prompt

    def validate(self, prompt: dict[str, Any]) -> None:
        errors: list[str] = []

        if Draft202012Validator:
            for e in Draft202012Validator(COMFY_PROMPT_SCHEMA).iter_errors(prompt):
                errors.append(f"schema: {list(e.path)} {e.message}")
        else:
            if not isinstance(prompt, dict) or not prompt:
                errors.append("schema: prompt must be non-empty object")
            for k, v in prompt.items():
                if not str(k).isdigit():
                    errors.append(f"schema: node id must be numeric string: {k}")
                if not isinstance(v, dict):
                    errors.append(f"schema: node {k} must be object")
                elif "class_type" not in v or "inputs" not in v:
                    errors.append(f"schema: node {k} missing class_type or inputs")

        if errors:
            raise FlowValidationError(errors)

        edges: dict[str, set[str]] = {nid: set() for nid in prompt}
        preds: dict[str, set[str]] = {nid: set() for nid in prompt}

        for nid, node in prompt.items():
            cls = node.get("class_type")
            inputs = node.get("inputs", {})

            if cls not in self.allowed_classes:
                errors.append(
                    f"node {nid}: hallucinated class_type '{cls}', "
                    f"not found in ComfyUI /object_info"
                )
                continue

            allowed_inputs = self._allowed_inputs(cls)
            if allowed_inputs:
                for key in inputs.keys():
                    if key not in allowed_inputs:
                        errors.append(
                            f"node {nid} {cls}: unknown input '{key}', "
                            f"allowed={sorted(allowed_inputs)}"
                        )

            for ref_id, out_idx, path in self._iter_refs(inputs):
                if ref_id not in prompt:
                    errors.append(f"node {nid}: broken ref at {path}: node {ref_id} not found")
                    continue
                if ref_id == nid:
                    errors.append(f"node {nid}: self reference at {path}")
                    continue
                if not isinstance(out_idx, int) or out_idx < 0:
                    errors.append(f"node {nid}: invalid output index at {path}: {out_idx}")
                    continue

                max_outputs = self._output_count(prompt[ref_id].get("class_type"))
                if max_outputs is not None and out_idx >= max_outputs:
                    errors.append(
                        f"node {nid}: output index {out_idx} out of range for "
                        f"{ref_id}:{prompt[ref_id].get('class_type')}, max={max_outputs - 1}"
                    )

                edges[ref_id].add(nid)
                preds[nid].add(ref_id)

        if errors:
            raise FlowValidationError(errors)

        self._check_acyclic(edges)
        self._check_reaches_sink(prompt, preds)

    @staticmethod
    def extract_json(text: str) -> dict[str, Any]:
        text = text.strip()

        fenced = re.search(r"```(?:json)?\s*(.*?)```", text, re.S)
        if fenced:
            text = fenced.group(1).strip()

        try:
            obj = json.loads(text)
        except json.JSONDecodeError:
            start = text.find("{")
            end = text.rfind("}")
            if start < 0 or end < 0 or end <= start:
                raise FlowValidationError(["invalid JSON: no JSON object found"])
            obj = json.loads(text[start : end + 1])

        if not isinstance(obj, dict):
            raise FlowValidationError(["invalid JSON: root must be object"])

        return obj

    def _allowed_inputs(self, class_type: str) -> set[str]:
        info = self.node_catalog.get(class_type) or {}
        inputs = info.get("input") or {}
        allowed = set()
        for section in ("required", "optional", "hidden"):
            allowed |= set((inputs.get(section) or {}).keys())
        return allowed

    def _output_count(self, class_type: str | None) -> int | None:
        if not class_type:
            return None
        info = self.node_catalog.get(class_type) or {}
        outputs = info.get("output")
        if isinstance(outputs, list):
            return len(outputs)
        return None

    def _iter_refs(self, value: Any, path: str = "inputs"):
        """
        ComfyUI ref shape: ["node_id", output_index]
        """
        if (
            isinstance(value, list)
            and len(value) == 2
            and isinstance(value[0], (str, int))
            and str(value[0]).isdigit()
            and isinstance(value[1], int)
        ):
            yield str(value[0]), value[1], path
            return

        if isinstance(value, dict):
            for k, v in value.items():
                yield from self._iter_refs(v, f"{path}.{k}")
        elif isinstance(value, list):
            for i, v in enumerate(value):
                yield from self._iter_refs(v, f"{path}[{i}]")

    def _check_acyclic(self, edges: dict[str, set[str]]) -> None:
        color: dict[str, int] = {}

        def dfs(n: str):
            color[n] = 1
            for m in edges[n]:
                if color.get(m) == 1:
                    raise FlowValidationError([f"topology: cycle detected at {n} -> {m}"])
                if color.get(m, 0) == 0:
                    dfs(m)
            color[n] = 2

        for n in edges:
            if color.get(n, 0) == 0:
                dfs(n)

    def _check_reaches_sink(self, prompt: dict[str, Any], preds: dict[str, set[str]]) -> None:
        sinks = [
            nid for nid, node in prompt.items()
            if node.get("class_type") in self.sink_classes
        ]

        if not sinks:
            raise FlowValidationError([
                f"topology: no output/sink node found, expected one of {sorted(self.sink_classes)}"
            ])

        reachable: set[str] = set()

        def walk_back(n: str):
            if n in reachable:
                return
            reachable.add(n)
            for p in preds[n]:
                walk_back(p)

        for sink in sinks:
            walk_back(sink)

        unused = sorted(set(prompt.keys()) - reachable, key=int)
        if unused:
            raise FlowValidationError([
                f"topology: disconnected / unused nodes not reaching output: {unused}"
            ])


def build_flow_prompt(nl: str, node_catalog: dict[str, Any]) -> str:
    """
    防幻觉 Prompt：
    - 只允许使用 catalog 中存在的 class_type
    - 禁止自然语言解释
    - 禁止 Markdown
    - 禁止发明 node/input 名称
    """

    compact_catalog = {}
    for cls, info in node_catalog.items():
        inputs = info.get("input", {})
        compact_catalog[cls] = {
            "required": list((inputs.get("required") or {}).keys()),
            "optional": list((inputs.get("optional") or {}).keys()),
            "output": info.get("output", []),
        }

    return f"""
你是 ComfyUI 工作流 JSON 编译器。

用户需求：
{nl}

硬规则：
1. 只输出 ComfyUI API Prompt JSON，不要解释，不要 Markdown。
2. JSON 根对象必须是 node_id -> node。
3. node_id 必须是数字字符串，例如 "1"、"2"。
4. 每个节点格式必须是：
   {{
     "class_type": "必须来自 ALLOWED_NODE_CATALOG 的精确名称",
     "inputs": {{}}
   }}
5. 禁止发明 class_type。
6. 禁止发明 input 名称。
7. 节点连接必须使用 ComfyUI 引用格式：["上游node_id", output_index]。
8. workflow 必须连到输出节点，例如 SaveImage 或 PreviewImage。
9. 不要生成孤儿节点，不要生成断开的子图。
10. 如果某个能力没有对应节点，使用最接近的已知节点，不要自造节点。

ALLOWED_NODE_CATALOG:
{json.dumps compact_catalog if False else json.dumps(compact_catalog, ensure_ascii=False, indent=2)}

只输出 JSON：
""".strip()


def build_repair_prompt(
    nl: str,
    bad_json: str,
    errors: list[str],
    node_catalog: dict[str, Any],
) -> str:
    return f"""
你上一次生成的 ComfyUI workflow JSON 校验失败。

原始用户需求：
{nl}

校验错误：
{json.dumps(errors, ensure_ascii=False, indent=2)}

错误 JSON：
{bad_json}

修复要求：
1. 只修 JSON。
2. 不要解释。
3. 不要 Markdown。
4. class_type 必须来自允许节点。
5. 修复断链、幻觉节点、非法 input、无输出节点、孤儿节点。
6. 输出完整可执行 ComfyUI API Prompt JSON。

允许节点：
{json.dumps(list(node_catalog.keys()), ensure_ascii=False)}

只输出 JSON：
""".strip()


class FlowCompiler:
    """
    最小 NL -> ComfyUI Prompt JSON 编译器。

    llm 参数是一个函数：
        llm(prompt: str) -> str

    node_catalog 建议来自：
        GET http://127.0.0.1:8188/object_info
    """

    def __init__(
        self,
        llm: Callable[[str], str],
        node_catalog: dict[str, Any],
        max_retries: int = 2,
    ):
        self.llm = llm
        self.node_catalog = node_catalog
        self.validator = FlowValidator(node_catalog)
        self.max_retries = max_retries

    def compile(self, nl: str) -> dict[str, Any]:
        prompt = build_flow_prompt(nl, self.node_catalog)
        last_text = ""

        for attempt in range(self.max_retries + 1):
            text = self.llm(prompt)
            last_text = text

            try:
                flow = self.validator.parse_and_validate(text)
                return flow

            except FlowValidationError as e:
                if attempt >= self.max_retries:
                    raise

                prompt = build_repair_prompt(
                    nl=nl,
                    bad_json=text,
                    errors=e.errors,
                    node_catalog=self.node_catalog,
                )

        raise FlowValidationError([f"compile failed: {last_text}:
                if attempt >= self.max_retries:
                    raise

                prompt"])

import requests
from flow_compiler import FlowCompiler

def llm(prompt: str) -> str:
    # 接你的 BrainProvider / OpenAI / DeepSeek / Ollama 都行
    return brain.generate(prompt)

object_info = requests.get("http://127.0.0.1:8188/object_info", timeout=5).json()

compiler = FlowCompiler(
    llm=llm,
    node_catalog=object_info,
    max_retries=2,
)

workflow = compiler.compile("生成一张赛博城市雨夜，512x768，保存图片")

核心重构边界：

NL 输入
  ↓
Prompt 模板只给真实 node catalog
  ↓
LLM 生成候选 JSON
  ↓
FlowValidator 强校验
  ↓
失败则把 errors 回灌给 LLM 修复
  ↓
成功后才提交给 ComfyUI

这样三个问题会被硬挡住：

非法 JSON        → extract_json + JSON Schema
幻觉 node 名称   → class_type 必须存在于 /object_info
拓扑不连通       → 所有节点必须能反向追溯到 SaveImage / PreviewImage 等 sink
