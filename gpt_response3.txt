# hardware_profiler.py
from __future__ import annotations

import os
import platform
import subprocess
from dataclasses import dataclass
from typing import Optional

import psutil


@dataclass
class HardwareProfile:
    gpu_name: Optional[str]
    gpu_vram_gb: float
    gpu_free_vram_gb: Optional[float]
    ram_gb: float
    cpu_name: Optional[str]
    cpu_cores: int
    cuda_available: bool
    device_type: str
    os_name: str


class HardwareProfiler:
    def detect(self) -> HardwareProfile:
        gpu_name = None
        total_vram = 0.0
        free_vram = None
        cuda_available = False
        device_type = "cpu"

        try:
            import torch

            if torch.cuda.is_available():
                cuda_available = True
                device_type = "cuda"
                gpu_name = torch.cuda.get_device_name(0)

                props = torch.cuda.get_device_properties(0)
                total_vram = props.total_memory / 1024**3

                free, total = torch.cuda.mem_get_info(0)
                free_vram = free / 1024**3
        except Exception:
            pass

        if not gpu_name:
            info = self._detect_nvidia_smi()
            if info:
                gpu_name = info["name"]
                total_vram = info["total_vram_gb"]
                free_vram = info["free_vram_gb"]
                cuda_available = True
                device_type = "cuda"

        return HardwareProfile(
            gpu_name=gpu_name,
            gpu_vram_gb=round(total_vram, 2),
            gpu_free_vram_gb=round(free_vram, 2) if free_vram is not None else None,
            ram_gb=round(psutil.virtual_memory().total / 1024**3, 2),
            cpu_name=platform.processor() or None,
            cpu_cores=os.cpu_count() or 1,
            cuda_available=cuda_available,
            device_type=device_type,
            os_name=platform.system(),
        )

    def _detect_nvidia_smi(self) -> Optional[dict]:
        try:
            cmd = [
                "nvidia-smi",
                "--query-gpu=name,memory.total,memory.free",
                "--format=csv,noheader,nounits",
            ]
            out = subprocess.check_output(cmd, text=True, timeout=5).strip()
            first = out.splitlines()[0]
            name, total, free = [x.strip() for x in first.split(",")]

            return {
                "name": name,
                "total_vram_gb": float(total) / 1024,
                "free_vram_gb": float(free) / 1024,
            }
        except Exception:
            return None

from dataclasses import dataclass
from typing import Tuple

from hardware_profiler import HardwareProfile


@dataclass
class RuntimeBudget:
    tier: str

    usable_vram_gb: float
    max_resolution: Tuple[int, int]
    max_batch_size: int

    allow_sd15: bool
    allow_sdxl: bool
    allow_refiner: bool
    allow_upscale: bool
    allow_controlnet: bool
    allow_video: bool

    use_tiled_vae: bool
    use_cpu_offload: bool
    precision: str


class RuntimeBudgetEngine:
    def build(self, hw: HardwareProfile) -> RuntimeBudget:
        total = hw.gpu_vram_gb or 0.0

        # 给系统、浏览器、ComfyUI 运行时留余量
        usable = max(0.0, total - 1.5)

        if total < 4:

                tier="cpu_or_tiny",

                max_resolution=(512, 768),
                max_batch_size=1,

                allow_video=False,

                use_cpu_offload=True,
                precision="fp16",
            )

        if total < 8:
            return RuntimeBudget(
                tier="low_6gb",
                usable_vram_gb=usable,
                max_resolution=(768, 1152),
                max_batch_size=1,
                allow_sd15=True,
                allow_sdxl=False,
                allow_refiner=False,
                allow_upscale=False,
                allow_controlnet=True,
                allow_video=False,

五、Planner 的目标：在约束内最大化质量

这一步很关键。

以前的降级链是：

高级 → 中级 → 低级

现在应该变成：

从所有候选中筛掉不可运行的
然后在剩下的里面选质量最高的

也就是一个约束优化问题：

maximize quality_score
subject to:
  blueprint.min_vram <= usable_vram
  required_nodes installed
  required_models installed
  task_type matches
  output_ratio supported
  estimated_peak_vram <= budget

# hardware_aware_planner.py
from typing import List, Optional

from runtime_budget import RuntimeBudget


class HardwareAwarePlanner:
    def select_best(
        self,
        task_type: str,
        blueprints: List[BlueprintContract],
        budget: RuntimeBudget,
        installed_features: dict,
    ) -> Optional[BlueprintContract]:
        candidates = []

        for bp in blueprints:
            if bp.task_type != task_type:
                continue

            if not self._fits_budget(bp, budget):
                continue

            if not self._fits_environment(bp, installed_features):
                continue

            candidates.append(bp)

        if not candidates:
            return None

        return sorted(
            candidates,
            key=lambda x: (
                x.quality_score,
                -x.recommended_vram_gb,
            ),
            reverse=True,
        )[0]

    def _fits_budget(self, bp: BlueprintContract, budget: RuntimeBudget) -> bool:
        if bp.min_vram_gb > budget.usable_vram_gb:
            return False

        if bp.requires_sdxl and not budget.allow_sdxl:
            return False

        if bp.requires_refiner and not budget.allow_refiner:
            return False

        if bp.requires_upscale and not budget.allow_upscale:
            return False

        if bp.requires_controlnet and not budget.allow_controlnet:
            return False

        if bp.requires_video and not budget.allow_video:
            return False

        return True

    def _fits_environment(self, bp: BlueprintContract, installed: dict) -> bool:
        if bp.requires_sdxl and not installed.get("has_sdxl_model"):
            return False

        if bp.requires_refiner and not installed.get("has_refiner_model"):
            return False

        if bp.requires_upscale and not installed.get("has_upscale_model"):
            return False

        if bp.requires_controlnet and not installed.get("has_controlnet"):
            return False

        if bp.requires_video and not installed.get("has_video_model"):
            return False

        return True

class HardwareAwareParameterPolicy:
    def apply(self, spec, budget: RuntimeBudget):
        w, h = self._fit_ratio_to_budget(
            aspect_ratio=spec.aspect_ratio,
            max_resolution=budget.max_resolution,
        )

        spec.width = w
        spec.height = h
        spec.batch_size = min(spec.batch_size or 1, budget.max_batch_size)

        if budget.tier in ["cpu_or_tiny", "low_6gb"]:
            spec.steps = min(spec.steps or 24, 24)
            spec.cfg = min(spec.cfg or 7.0, 7.5)
            spec.enable_tiled_vae = True
            spec.enable_refiner = False
            spec.enable_upscale = False

        elif budget.tier in ["mid_8gb", "upper_12gb"]:
            spec.steps = min(spec.steps or 30, 32)
            spec.cfg = min(spec.cfg or 7.0, 8.0)
            spec.enable_tiled_vae = True
            spec.enable_refiner = False

        else:
            spec.steps = min(spec.steps or 36, 40)
            spec.cfg = min(spec.cfg or 7.0, 8.5)
            spec.enable_tiled_vae = budget.use_tiled_vae

        return spec

    def _fit_ratio_to_budget(self, aspect_ratio: str, max_resolution: tuple[int, int]):
        max_w, max_h = max_resolution

        if aspect_ratio == "9:16":
            h = max_h
            w = int(h * 9 / 16)
            w = self._round8(min(w, max_w))
            h = self._round8(h)
            return w, h

        if aspect_ratio == "16:9":
            w = max_w
            h = int(w * 9 / 16)
            return self._round8(w), self._round8(h)

        size = min(max_w, max_h)
        return self._round8(size), self._round8(size)

    def _round8(self, x: int) -> int:
        return max(8, int(round(x / 8) * 8))
七、动态降级链如何生成？

def build_dynamic_fallback_chain(task_type, blueprints, budget, installed):
    runnable = []

    for bp in blueprints:
        if bp.task_type != task_type:
            continue

        if fits_environment(bp, installed):
            runnable.append(bp)

    # 先按质量从高到低排
    runnable.sort(key=lambda x: x.quality_score, reverse=True)

    # 再标记是否硬件可跑
    safe = [bp for bp in runnable if fits_budget(bp, budget)]
    risky = [bp for bp in runnable if not fits_budget(bp, budget)]

    return safe, risky

@dataclass
class VramRiskReport:
    passed: bool
    estimated_peak_gb: float
    usable_vram_gb: float
    risk_level: str
    suggestions: list[str]


class VramGate:
    def check(self, workflow, budget: RuntimeBudget) -> VramRiskReport:
        estimated = self.estimate_peak_vram(workflow)

        if estimated <= budget.usable_vram_gb * 0.85:
            return VramRiskReport(True, estimated, budget.usable_vram_gb, "low", [])

        if estimated <= budget.usable_vram_gb:
            return VramRiskReport(
                True,
                estimated,
                budget.usable_vram_gb,
                "medium",
                ["接近显存上限，建议启用 tiled VAE 或降低分辨率。"],
            )

        return VramRiskReport(
            False,
            estimated,
            budget.usable_vram_gb,
            "high",
            [
                "预计会爆显存。",
                "建议降低分辨率。",
                "关闭 Refiner。",
                "关闭 Upscale。",
                "切换到轻量模型。",
            ],
        )

    def estimate_peak_vram(self, workflow) -> float:
        # MVP 可以先用规则估算，后续用真实运行数据学习修正
        node_count = len(workflow)
        has_sdxl = self._has_model_keyword(workflow, ["sdxl", "xl"])
        has_refiner = self._has_model_keyword(workflow, ["refiner"])
        has_upscale = self._has_node_keyword(workflow, ["upscale"])
        has_video = self._has_node_keyword(workflow, ["video", "ltx", "wan"])

        base = 3.5

        if has_sdxl:
            base += 4.0
        if has_refiner:
            base += 3.0
        if has_upscale:
            base += 2.0
        if has_video:
            base += 8.0

        base += max(0, node_count - 10) * 0.05
        return round(base, 2)

    def _has_model_keyword(self, workflow, keywords):
        text = str(workflow).lower()
        return any(k.lower() in text for k in keywords)

    def _has_node_keyword(self, workflow, keywords):
        text = str(workflow).lower()
        return any(k.lower() in text for k in keywords)

最终架构定位

加上硬件感知后，ComfyFlow Compiler 的真实定位应该是：

Hardware-aware Expert Workflow Compiler

它不是简单的：

自然语言 → workflow

而是：

自然语言
  + 用户零基础
  + 本机硬件
  + 已装模型
  + 已装节点
  + 质量目标
  ↓
当前机器能跑的最高级 ComfyUI 工作流

最终一句话：

硬件感知能力应该作为 Recipe/Blueprint 选择之前的“预算层”存在。系统先检测硬件和环境，生成 RuntimeBudget，再在预算约束内最大化工作流质量；降级链不再固定，而是由硬件、模型、节点、任务类型共同动态生成。
