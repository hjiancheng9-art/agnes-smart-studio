# comfyflow/user_facing.py
from dataclasses import dataclass, field
from typing import Any, Literal


@dataclass
class UserFacingMessage:
    level: Literal["info", "warning", "error"]
    title: str
    message: str
    suggestion: str | None = None


@dataclass
class UserFacingResult:
    success: bool
    title: str
    summary: str
    workflow: dict[str, Any] | None = None
    messages: list[UserFacingMessage] = field(default_factory=list)
    technical_report: dict[str, Any] = field(default_factory=dict)

    def to_dict(self) -> dict[str, Any]:
        return {
            "success": self.success,
            "title": self.title,
            "summary": self.summary,
            "workflow": self.workflow,
            "messages": [m.__dict__ for m in self.messages],
            "technical_report": self.technical_report,
        }

def translate_error(code: str, detail: str | None = None) -> dict:
    base = ERROR_TRANSLATIONS.get(code, {
        "title": "工作流生成失败",
        "message": "系统遇到了未分类错误。",
        "suggestion": "请尝试降低质量模式，或切换基础工作流。",
    })

    return {
        "level": "error",
        "title": base["title"],
        "message": base["message"] if not detail else f'{base["message"]}（{detail}）',
        "suggestion": base["suggestion"],
    }
