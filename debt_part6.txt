# comfyflow/experiment.py
import hashlib


def choose_variant(user_or_task_id: str, experiment_name: str, variants: list[str]) -> str:
    key = f"{experiment_name}:{user_or_task_id}"
    value = int(hashlib.sha256(key.encode("utf-8")).hexdigest(), 16)
    return variants[value % len(variants)]
