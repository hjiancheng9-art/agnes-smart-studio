1.1 TaskSpec：用户任务规格

TaskSpec 表示“用户想做什么”，不出现 ComfyUI 节点名。

Python
运行
from dataclasses import dataclass, field
from typing import Any, Literal


TaskType = Literal[
    "txt2img",
    "img2img",
    "inpaint",
    "controlnet",
    "ip_adapter",
    "upscale",
    "image_to_video",
    "text_to_video",
    "lora_training",
    "pipeline",
]


@dataclass
class AssetRef:
    asset_id: str
    type: Literal["image", "video", "mask", "lora", "model", "text"]
    uri: str
    role: str | None = None
    metadata: dict[str, Any] = field(default_factory=dict)


@dataclass
class TaskSpec:
    task_id: str
    task_type: TaskType

    prompt: str | None = None
    negative_prompt: str | None = None

    input_assets: list[AssetRef] = field(default_factory=list)
    output_types: list[str] = field(default_factory=list)

    aspect_ratio: str | None = None
    quality_target: Literal["fast", "balanced", "high"] = "balanced"
    style_tags: list[str] = field(default_factory=list)

    crux_stage: str | None = None
    preferred_model_family: str | None = None

    user_params: dict[str, Any] = field(default_factory=dict)
    constraints: dict[str, Any] = field(default_factory=dict)
