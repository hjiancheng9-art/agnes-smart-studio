@dataclass
class RuntimeBudget:
    tier: str

    usable_vram_gb: float
    max_resolution: Tuple[int, int]
    max_batch_size: int

    allow_sd15: bool
    allow_sdxl: bool
    allow_refiner: bool
    allow_upscale: bool
    allow_controlnet: bool
    allow_video: bool

    use_tiled_vae: bool
    use_cpu_offload: bool
    precision: str


class RuntimeBudgetEngine:
    def build(self, hw: HardwareProfile) -> RuntimeBudget:
        total = hw.gpu_vram_gb or 0.0

        # 给系统、浏览器、ComfyUI 运行时留余量
        usable = max(0.0, total - 1.5)

        if total < 4:
            return RuntimeBudget(
                tier="cpu_or_tiny",
                usable_vram_gb=usable,
                max_resolution=(512, 768),
                max_batch_size=1,
                allow_sd15=True,
                allow_sdxl=False,
                allow_refiner=False,
                allow_upscale=False,
                allow_controlnet=False,
                allow_video=False,
                use_tiled_vae=True,
                use_cpu_offload=True,
                precision="fp16",
            )

                tier="low_6gb",
