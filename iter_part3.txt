
from collections import Counter
from dataclasses import dataclass


@dataclass
class MinedBlueprint:
    id: str
    category: str
    required_nodes: list[str]
    optional_nodes: list[str]
    common_edges: list[tuple[str, str]]
    source_workflow_count: int
    estimated_node_count: int
    confidence_score: float


def mine_category_blueprint(workflows: list[dict], category: str) -> MinedBlueprint:
    selected = [w for w in workflows if w["category"] == category]
    total = len(selected)

    node_counter = Counter()
    edge_counter = Counter()
    node_counts = []

    for w in selected:
        node_types = w["node_types"]
        edges = w.get("edges_by_type", [])

        node_counts.append(len(node_types))

        for t in set(node_types):
            node_counter[t] += 1

        for edge in edges:
            edge_counter[tuple(edge)] += 1

    required_nodes = [
        node for node, count in node_counter.items()
        if count / total >= 0.6
    ]

    optional_nodes = [
        node for node, count in node_counter.items()
        if 0.25 <= count / total < 0.6
    ]

    common_edges = [
        edge for edge, count in edge_counter.items()
        if count / total >= 0.3
    ]

    confidence = min(1.0, len(selected) / 20) * min(1.0, len(required_nodes) / 8)

    return MinedBlueprint(
        id=f"{category}_production_mined",
        category=category,
