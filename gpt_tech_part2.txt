@dataclass
class GPUInfo:
    gpu_name: Optional[str]
    vram_gb: float
    cuda_available: bool
    source: str
    error: Optional[str] = None


def detect_gpu() -> GPUInfo:
    """
    Windows / Linux 都可用。
    优先级：
      1. pynvml
      2. nvidia-smi
      3. torch.cuda
      4. fallback: 无 NVIDIA GPU

    返回：
      gpu_name: str | None
      vram_gb: float
      cuda_available: bool
    """

    # 1. pynvml: 最可靠，适合 Windows + NVIDIA
    nvml_result = _detect_gpu_with_pynvml()
    if nvml_result.cuda_available:
        return nvml_result

    smi_result = _detect_gpu_with_nvidia_smi()
    if smi_result.cuda_available:
        return smi_result

        return torch_result
