def classify_production_intent(text: str) -> str:
    t = text.lower()

    if any(x in t for x in ["动作迁移", "动作参考", "motion transfer", "animate"]):
        return "wan_action_transfer"

    if any(x in t for x in ["换人", "人物替换", "换脸", "character replace"]):
        return "wan_character_replace"

    if any(x in t for x in ["上传音频", "音频驱动", "audio", "对口型", "lip sync"]):
        return "ltx_i2v_audio"

    if any(x in t for x in ["图生视频", "i2v", "image to video"]):
        return "ltx_i2v"

    if any(x in t for x in ["视频", "video", "镜头", "运镜"]):
        return "ltx_t2v"

    if any(x in t for x in ["编辑图片", "换装", "改图", "image edit", "kontext"]):
        return "flux_image_edit"

    if any(x in t for x in ["flux", "高质量", "电影感", "真实感"]):
        return "flux_txt2img"

    return "sd_txt2img"
升级 2：蓝图必须支持 multi-stage pipeline

真实高级工作流不是一个 graph，而是多个 stage：

stage 1: 输入 / 预处理
stage 2: 图像生成 / 编辑

stage 4: 音频 / latent 处理
stage 5: 输出合成
