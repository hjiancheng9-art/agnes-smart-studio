   1: """Message pane — scrollable chat buffer with mouse wheel, keyboard, and auto-scroll.
   2: 
   3: Uses a custom ScrollingWindow subclass to prevent prompt_toolkit 3.x from
   4: resetting vertical_scroll every frame (known issue in _scroll() method).
   5: """
   6: 
   7: from __future__ import annotations
   8: 
   9: import threading
  10: 
  11: from prompt_toolkit.formatted_text import FormattedText
  12: from prompt_toolkit.layout import Window
  13: from prompt_toolkit.layout.controls import FormattedTextControl
  14: from prompt_toolkit.layout.dimension import Dimension
  15: from prompt_toolkit.mouse_events import MouseEventType
  16: 
  17: _ROLE_FORMATS = {
  18:     "user": ("class:message-user", "You"),
  19:     "crux": ("class:message-crux", "CRUX"),
  20:     "assistant": ("class:message-crux", "CRUX"),
  21:     "info": ("class:message-info", "--"),
  22:     "error": ("class:message-error", "!!"),
  23: }
  24: 
  25: # Sentinel value: setting vertical_scroll to a very large number makes
  26: # ptk clamp it to the actual max, guaranteeing bottom-scroll regardless
  27: # of content height changes.
  28: _SCROLL_BOTTOM = 999999
  29: 
  30: # Scroll speed
  31: _SCROLL_LINE = 3   # lines per mouse wheel tick
  32: _SCROLL_PAGE_FACTOR = 0.85  # fraction of visible height for PageUp/PageDown
  33: 
  34: 
  35: class _ScrollingWindow(Window):
  36:     """Window subclass that overrides _scroll() to prevent ptk 3.x from
  37:     resetting vertical_scroll to 0 on every render frame.
  38: 
  39:     ptk 3.0.x recalculates vertical_scroll in _scroll() based on cursor
  40:     position, which is always (0,0) for FormattedTextControl. This means
  41:     any manual scroll position is lost on the next frame unless we no-op
  42:     the internal scroll calculation.
  43: 
  44:     Also correctly handles auto-scroll-to-bottom when wrap_lines=True:
  45:     computes both vertical_scroll (content line index) and vertical_scroll_2
  46:     (visual row offset within that line) so the bottom of wrapped content
  47:     is visible even when total wrapped height exceeds window height.
  48:     """
  49: 
  50:     def _scroll(self, ui_content, width, height):
  51:         # Let the parent calculate, but then restore our saved position
  52:         saved = self.vertical_scroll
  53:         super()._scroll(ui_content, width, height)
  54:         if saved >= _SCROLL_BOTTOM:
  55:             # Pinned to bottom: compute actual scroll position accounting
  56:             # for line wrapping (vertical_scroll_2 skips visual rows).
  57:             total_wrapped = 0
  58:             for i in range(ui_content.line_count if ui_content else 0):
  59:                 total_wrapped += ui_content.get_height_for_line(
  60:                     i, width, self.get_line_prefix
  61:                 )
  62: 
  63:             if total_wrapped <= height:
  64:                 # Content fits in window — no scrolling needed
  65:                 self.vertical_scroll = 0
  66:                 self.vertical_scroll_2 = 0
  67:             else:
  68:                 # Content exceeds window height — find the right content
  69:                 # line and intra-line offset to show the bottom 'height'
  70:                 # visual rows. Iterate forward from top: skip accumulates
  71:                 # until we find the first visible line.
  72:                 skip = total_wrapped - height
  73:                 for lineno in range(ui_content.line_count):
  74:                     line_h = ui_content.get_height_for_line(
  75:                         lineno, width, self.get_line_prefix
  76:                     )
  77:                     if line_h <= skip:
  78:                         skip -= line_h
  79:                     else:
  80:                         self.vertical_scroll = lineno
  81:                         self.vertical_scroll_2 = skip
  82:                         break
  83:         elif saved > 0:
  84:             self.vertical_scroll = saved
  85: 
  86: 
  87: class _MessagePaneControl(FormattedTextControl):
  88:     """FormattedTextControl subclass with mouse scroll handler."""
  89: 
  90:     def __init__(self, pane, *args, **kwargs):
  91:         self._mp_pane = pane
  92:         super().__init__(*args, **kwargs)
  93: 
  94:     def mouse_handler(self, mouse_event):
  95:         from prompt_toolkit.application.current import get_app
  96:         if mouse_event.event_type == MouseEventType.SCROLL_UP:
  97:             self._mp_pane.scroll_up(lines=_SCROLL_LINE)
  98:             get_app().invalidate()
  99:             return None
 100:         if mouse_event.event_type == MouseEventType.SCROLL_DOWN:
 101:             self._mp_pane.scroll_down(lines=_SCROLL_LINE)
 102:             get_app().invalidate()
 103:             return None
 104:         return super().mouse_handler(mouse_event)
 105: 
 106: 
 107: class MessagePane:
 108:     """Scrollable chat message display with auto-scroll and manual override.
 109: 
 110:     Behavior:
 111:     - New messages auto-scroll to bottom when _pinned=True
 112:     - User scrolls up (PageUp / mouse wheel up) → unpin, stop auto-scrolling
 113:     - User scrolls to bottom (PageDown / mouse wheel down to end) → re-pin
 114:     - stream_start() always re-pins to show live streaming output
 115:     """
 116: 
 117:     def __init__(self) -> None:
 118:         self._lock = threading.Lock()
 119:         self._lines: list[tuple[str, str]] = []
 120:         self._stream_role = ""
 121:         self._stream_label = ""
 122:         self._stream_buffer = ""
 123:         self._pinned = True
 124:         self._empty_renderer = None  # type: ignore[assignment]
 125:         self._empty_render_cache = None
 126:         self._empty_render_cache_key = None
 127: 
 128:         def _render():
 129:             # Snapshot state atomically, render without lock
 130:             with self._lock:
 131:                 if not self._lines and not self._stream_buffer:
 132:                     if self._empty_renderer is not None:
 133:                         _empty_width = self._window.render_info.window_width if self._window.render_info else None
 134:                         _empty_key = _empty_width
 135:                         if self._empty_render_cache_key != _empty_key:
 136:                             self._empty_render_cache = self._empty_renderer()
 137:                             self._empty_render_cache_key = _empty_key
 138:                         return self._empty_render_cache
 139:                     return FormattedText([("class:message-info", "Type a message or /help for commands")])
 140:                 lines_snapshot = list(self._lines)
 141:                 stream_snapshot = self._stream_buffer
 142:             pieces = []
 143:             for style_class, text in lines_snapshot:
 144:                 pieces.append((style_class, text + "\n"))
 145:             if stream_snapshot:
 146:                 pieces.append(("class:message-crux", stream_snapshot))
 147:             return FormattedText(pieces)
 148: 
 149:         self._control = _MessagePaneControl(self, _render)
 150:         self._window = _ScrollingWindow(
 151:             content=self._control,
 152:             style="class:message-area",
 153:             wrap_lines=True,
 154:             height=Dimension(weight=1),
 155:             always_hide_cursor=True,
 156:             allow_scroll_beyond_bottom=False,
 157:             right_margins=[],
 158:         )
 159: 
 160:     # ── Public properties ────────────────────────────────────
 161: 
 162:     @property
 163:     def pane(self) -> Window:
 164:         return self._window
 165: 
 166:     @property
 167:     def line_count(self) -> int:
 168:         return len(self._lines) + (1 if self._stream_buffer else 0)
 169: 
 170:     def _window_height(self) -> int:
 171:         h = self._window.render_info.window_height if self._window.render_info else 0
 172:         return max(5, h - 1) if h > 0 else 20
 173: 
 174:     # ── Wrapped content helpers ──────────────────────────────
 175: 
 176:     def _wrapped_content_height(self) -> int:
 177:         """Total visual row count of all messages after line wrapping.
 178:         Falls back to unwrapped line_count when render_info is unavailable."""
 179:         ri = self._window.render_info
 180:         if ri is not None:
 181:             return ri.content_height
 182:         return self.line_count
 183: 
 184:     def _wrapped_window_height(self) -> int:
 185:         """Current window height in visual rows."""
 186:         ri = self._window.render_info
 187:         if ri is not None:
 188:             return ri.window_height
 189:         return self._window_height()
 190: 
 191:     def _current_visual_row(self) -> int:
 192:         """Estimate the topmost visible visual row from scroll state."""
 193:         vs = self._window.vertical_scroll
 194:         if vs >= _SCROLL_BOTTOM:
 195:             return max(0, self._wrapped_content_height() - self._wrapped_window_height())
 196:         ri = self._window.render_info
 197:         if ri is not None and ri.ui_content is not None:
 198:             uc = ri.ui_content
 199:             row = 0
 200:             for i in range(min(vs, uc.line_count)):
 201:                 row += uc.get_height_for_line(
 202:                     i, ri.window_width, self._window.get_line_prefix
 203:                 )
 204:             return row + self._window.vertical_scroll_2
 205:         # Fallback: clamp vs to valid range
 206:         return min(vs, max(0, self.line_count - self._window_height()))
 207: 
 208:     def _set_scroll_to_visual_row(self, target_row: int, *, clamp: bool = True) -> None:
 209:         """Set vertical_scroll and vertical_scroll_2 to show a given visual row
 210:         as the topmost visible row. When clamp=True, constrain to valid range."""
 211:         ri = self._window.render_info
 212:         if ri is None or ri.ui_content is None:
 213:             # Fallback: clamp to valid content range
 214:             max_line = max(0, self.line_count - self._window_height())
 215:             self._window.vertical_scroll = max(0, min(target_row, max_line))
 216:             self._window.vertical_scroll_2 = 0
 217:             return
 218: 
 219:         uc = ri.ui_content
 220:         width = ri.window_width
 221: 
 222:         if clamp:
 223:             max_row = max(0, self._wrapped_content_height() - self._wrapped_window_height())
 224:             target_row = max(0, min(target_row, max_row))
 225: 
 226:         if target_row <= 0:
 227:             self._window.vertical_scroll = 0
 228:             self._window.vertical_scroll_2 = 0
 229:             return
 230: 
 231:         # Walk through content lines to find which one contains target_row
 232:         accum = 0
 233:         for lineno in range(uc.line_count):
 234:             line_h = uc.get_height_for_line(
 235:                 lineno, width, self._window.get_line_prefix
 236:             )
 237:             if accum + line_h > target_row:
 238:                 self._window.vertical_scroll = lineno
 239:                 self._window.vertical_scroll_2 = target_row - accum
 240:                 return
 241:             accum += line_h
 242: 
 243:         # Past the end — pin to bottom
 244:         self._window.vertical_scroll = _SCROLL_BOTTOM
 245:         self._window.vertical_scroll_2 = 0
 246: 
 247:     # ── Scrolling ────────────────────────────────────────────
 248: 
 249:     def scroll_up(self, lines: int = _SCROLL_LINE) -> None:
 250:         cur = self._current_visual_row()
 251:         self._set_scroll_to_visual_row(cur - lines)
 252:         new_row = self._current_visual_row()
 253:         if new_row > 0:
 254:             self._pinned = False
 255: 
 256:     def scroll_down(self, lines: int = _SCROLL_LINE) -> None:
 257:         cur = self._current_visual_row()
 258:         max_row = max(0, self._wrapped_content_height() - self._wrapped_window_height())
 259:         target = cur + lines
 260:         if target >= max_row:
 261:             self._pinned = True
 262:             self._window.vertical_scroll = _SCROLL_BOTTOM
 263:         else:
 264:             self._set_scroll_to_visual_row(target, clamp=False)
 265:             self._pinned = False
 266: 
 267:     def scroll_page_up(self) -> None:
 268:         page = max(5, int(self._wrapped_window_height() * _SCROLL_PAGE_FACTOR))
 269:         cur = self._current_visual_row()
 270:         self._set_scroll_to_visual_row(cur - page)
 271:         new_row = self._current_visual_row()
 272:         if new_row > 0:
 273:             self._pinned = False
 274: 
 275:     def scroll_page_down(self) -> None:
 276:         page = max(5, int(self._wrapped_window_height() * _SCROLL_PAGE_FACTOR))
 277:         cur = self._current_visual_row()
 278:         max_row = max(0, self._wrapped_content_height() - self._wrapped_window_height())
 279:         target = cur + page
 280:         if target >= max_row:
 281:             self._pinned = True
 282:             self._window.vertical_scroll = _SCROLL_BOTTOM
 283:         else:
 284:             self._set_scroll_to_visual_row(target, clamp=False)
 285:             self._pinned = False
 286: 
 287:     def scroll_to_top(self) -> None:
 288:         self._window.vertical_scroll = 0
 289:         self._window.vertical_scroll_2 = 0
 290:         self._pinned = False
 291: 
 292:     def scroll_to_bottom(self) -> None:
 293:         self._pinned = True
 294:         self._window.vertical_scroll = _SCROLL_BOTTOM
 295:         self._window.vertical_scroll_2 = 0
 296: 
 297:     def _auto_scroll(self) -> None:
 298:         if self._pinned:
 299:             self._window.vertical_scroll = _SCROLL_BOTTOM
 300:             self._window.vertical_scroll_2 = 0
 301: 
 302:     # ── Message management ───────────────────────────────────
 303: 
 304:     def append_message(self, role: str, text: str) -> None:
 305:         if not isinstance(text, str):
 306:             text = str(text) if text is not None else ""
 307:         with self._lock:
 308:             self._end_stream()
 309:             fmt = _ROLE_FORMATS.get(role)
 310:             sc, label = fmt if fmt else ("", role)
 311:             self._lines.append((sc, f"[{label}] {text}"))
 312:             self._lines.append(("", ""))
 313:             self._auto_scroll()
 314: 
 315:     def stream_start(self, role: str, *, force_pin: bool = True) -> None:
 316:         with self._lock:
 317:             if force_pin:
 318:                 self._pinned = True
 319:             self._end_stream()
 320:             fmt = _ROLE_FORMATS.get(role)
 321:             sc, label = fmt if fmt else ("", role)
 322:             self._stream_role = sc
 323:             self._stream_label = label
 324:             self._stream_buffer = f"[{label}] "
 325: 
 326:     def stream_append(self, text: str) -> None:
 327:         from utils.unicode_safety import sanitize_text
 328:         text = sanitize_text(text)
 329:         with self._lock:
 330:             if self._stream_label:
 331:                 self._stream_buffer += text
 332:                 self._auto_scroll()
 333: 
 334:     def stream_end(self) -> None:
 335:         with self._lock:
 336:             self._end_stream()
 337: 
 338:     def _end_stream(self) -> None:
 339:         if self._stream_buffer:
 340:             self._lines.append((self._stream_role, self._stream_buffer))
 341:             self._lines.append(("", ""))
 342:             self._stream_buffer = ""
 343:             self._stream_role = ""
 344:             self._stream_label = ""
 345:             self._auto_scroll()
 346: 
 347:     def append_info(self, text: str) -> None:
 348:         self.append_message("info", text)
 349: 
 350:     def append_error(self, text: str) -> None:
 351:         self.append_message("error", text)
 352: 
 353:     def set_empty_renderer(self, renderer) -> None:
 354:         """Set a custom renderer for the empty state (no messages, no stream).
 355: 
 356:         When set, this callable (returning FormattedText) is used instead of
 357:         the default placeholder text. Pass None to restore default.
 358:         """
 359:         self._empty_renderer = renderer
 360:         self._empty_render_cache = None
 361:         self._empty_render_cache_key = None
 362:         self._empty_render_cache_key = None
 363: 
 364:     def invalidate_empty_cache(self) -> None:
 365:         self._empty_render_cache = None
 366:         self._empty_render_cache_key = None
 367: 
 368:     def clear(self) -> None:
 369:         with self._lock:
 370:             self._end_stream()
 371:             self._lines.clear()
 372:             self._window.vertical_scroll = 0
 373:             self._pinned = True
 374:         self._empty_render_cache = None
 375:         self._empty_render_cache_key = None
 376:         self._empty_render_cache_key = None
