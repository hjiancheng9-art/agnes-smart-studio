    t = text.lower()

    if any(x in t for x in ["动作迁移", "动作参考", "motion transfer", "animate"]):

from collections import Counter, defaultdict
from dataclasses import dataclass


@dataclass
class MinedPattern:
    category: str
    node_set: list[str]
    frequency: int
    confidence: float


def mine_required_nodes(workflows, category, min_ratio=0.5):
    selected = [w for w in workflows if w["category"] == category]
    total = len(selected)

    counter = Counter()
    for w in selected:
        for node_type in set(w["node_types"]):
            counter[node_type] += 1

    patterns = []
    for node_type, count in counter.items():
        ratio = count / total
        if ratio >= min_ratio:
            patterns.append((node_type, count, round(ratio, 3)))

    return sorted(patterns, key=lambda x: x[1], reverse=True)


def mine_cooccurrence(workflows, min_count=5):
    pair_counter = Counter()

    for w in workflows:
        types = sorted(set(w["node_types"]))
        for i in range(len(types)):
            for j in range(i + 1, len(types)):
                pair_counter[(types[i], types[j])] += 1

    return [
        {
            "a": a,
            "b": b,
            "count": c,
        }
        for (a, b), c in pair_counter.items()
        if c >= min_count
    ]


def suggest_blueprint_from_workflows(workflows, category):
    required = mine_required_nodes(workflows, category, min_ratio=0.5)
    common_pairs = mine_cooccurrence(
        [w for w in workflows if w["category"] == category],
        min_count=5,
    )

    return {
        "category": category,
        "required_nodes": [x[0] for x in required],
        "node_frequency": required,
        "common_pairs": common_pairs[:50],
    }
