def select_blueprint(task, hardware, env):
    candidates = retrieve_blueprints(task)

    candidates = filter_by_comfyui_version(candidates, env.comfyui_version)
    candidates = filter_by_registered_nodes(candidates, env.object_info)
    candidates = filter_by_installed_models(candidates, env.models)
    candidates = filter_by_vram_budget(candidates, hardware.vram_gb)
    candidates = rank_by_quality(candidates, task.quality_mode)

    return candidates[0]
