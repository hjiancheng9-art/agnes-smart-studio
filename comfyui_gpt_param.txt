# parameter_service.py

from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional


@dataclass
class ParamPatch:
    node_id: str
    node_type: str
    param: str
    old_value: Any
    new_value: Any
    reason: str


@dataclass
class ParamTuneResult:
    workflow: Dict[str, Any]
    patches: List[ParamPatch] = field(default_factory=list)
    warnings: List[str] = field(default_factory=list)


class ParameterService:
    """
    唯一参数入口：
    - 动态读取 ComfyUI /object_info
    - 提供参数解释
    - 在 execute 前做安全调优
    """

    def __init__(self, comfy_client=None):
        self.comfy_client = comfy_client
        self.node_catalog: Dict[str, Any] = {}

    def refresh_from_comfyui(self) -> None:
        if not self.comfy_client:
            return
        self.node_catalog = self.comfy_client.get_object_info() or {}

    def get_param_info(self, node_type: str, param_name: str) -> Dict[str, Any]:
        node_info = self.node_catalog.get(node_type, {})
        inputs = node_info.get("input", {})
        required = inputs.get("required", {})
        optional = inputs.get("optional", {})

        raw = required.get(param_name) or optional.get(param_name)

        return {
            "node_type": node_type,
            "param": param_name,
            "schema": raw,
            "exists": raw is not None,
        }

    def optimize_before_execute(
        self,
        workflow: Dict[str, Any],
        intent: Optional[str] = None,
        quality_mode: str = "balanced",
    ) -> ParamTuneResult:
        result = ParamTuneResult(workflow=workflow)

        nodes = workflow.get("nodes") or workflow

        for node_id, node in nodes.items():
            node_type = node.get("class_type")
            inputs = node.get("inputs", {})

            if node_type == "KSampler":
                self._tune_ksampler(node_id, node_type, inputs, quality_mode, result)

            if node_type == "EmptyLatentImage":
                self._tune_latent(node_id, node_type, inputs, quality_mode, result)

        return result

    def _patch(
        self,
        result: ParamTuneResult,
        node_id: str,
        node_type: str,
        inputs: Dict[str, Any],
        param: str,
        new_value: Any,
        reason: str,
    ) -> None:
        old_value = inputs.get(param)
        if old_value == new_value:
            return

        inputs[param] = new_value
        result.patches.append(
            ParamPatch(
                node_id=node_id,
                node_type=node_type,
                param=param,
                old_value=old_value,
                new_value=new_value,
                reason=reason,
            )
        )

    def _tune_ksampler(
        self,
        node_id: str,
        node_type: str,
        inputs: Dict[str, Any],
        quality_mode: str,
        result: ParamTuneResult,
    ) -> None:
        steps = inputs.get("steps")

        if quality_mode == "fast":
            target_steps = 20
        elif quality_mode == "quality":
            target_steps = 32
        else:
            target_steps = 24

        if isinstance(steps, int):
            if steps < 12:
                self._patch(result, node_id, node_type, inputs, "steps", target_steps, "steps 太低，提升基础质量")
            elif steps > 60:
                self._patch(result, node_id, node_type, inputs, "steps", 36, "steps 过高，降低无效耗时")

        cfg = inputs.get("cfg")
        if isinstance(cfg, (int, float)):
            if cfg < 3:
                self._patch(result, node_id, node_type, inputs, "cfg", 6.5, "cfg 太低，提示词约束不足")
            elif cfg > 14:
                self._patch(result, node_id, node_type, inputs, "cfg", 8.0, "cfg 太高，容易过饱和或崩坏")

    def _tune_latent(
        self,
        node_id: str,
        node_type: str,
        inputs: Dict[str, Any],
        quality_mode: str,
        result: ParamTuneResult,
    ) -> None:
        width = inputs.get("width")
        height = inputs.get("height")

        if isinstance(width, int) and width % 8 != 0:
            self._patch(result, node_id, node_type, inputs, "width", width // 8 * 8, "宽度不是 8 的倍数")

        if isinstance(height, int) and height % 8 != 0:
            self._patch(result, node_id, node_type, inputs, "height", height // 8 * 8, "高度不是 8 的倍数")

# agent.py

class ComfyUIAgent:
    def __init__(self, *args, **kwargs):
        self._parameter_service = None
        self.flow_compiler = None
        self.validator = None
        self.executor = None
        self.comfy_client = None

    @property
    def parameter_service(self):
        if self._parameter_service is None:
            from parameter_service import ParameterService
            self._parameter_service = ParameterService(comfy_client=self.comfy_client)
            self._parameter_service.refresh_from_comfyui()
        return self._parameter_service

    def execute(self, workflow, *, auto_tune=True, quality_mode="balanced", intent=None):
        # 1. 执行前先校验原始 workflow
        valid = self.validator.validate(workflow)
        if not valid.ok:
            return {
                "success": False,
                "stage": "validate_before_tune",
                "error": valid.error,
            }

        tune_report = None

        # 2. 安全参数调优
        if auto_tune:
            tune_report = self.parameter_service.optimize_before_execute(
                workflow,
                intent=intent,
                quality_mode=quality_mode,
            )
            workflow = tune_report.workflow

            # 3. 调优后必须再次校验
            valid = self.validator.validate(workflow)
            if not valid.ok:
                return {
                    "success": False,
                    "stage": "validate_after_tune",
                    "error": valid.error,
                    "tune_report": tune_report,
                }

        # 4. 执行
        result = self.executor.execute(workflow)

        return {
            "success": result.success,
            "result": result,
            "tune_report": tune_report,
        }
