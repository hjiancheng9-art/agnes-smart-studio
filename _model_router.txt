851|    @staticmethod
852|    def classify_prompt(prompt: str) -> str:
853|        """Classify a user prompt into model tier: 'light', 'pro', or 'reasoner'.
854|
855|        Pure heuristics — no extra LLM call. Used by main chat auto-model mode.
856|        Returns one of: 'light', 'pro', 'reasoner'. Never raises.
857|        """
858|        if not prompt or not isinstance(prompt, str):
859|            return "pro"
860|
861|        stripped = prompt.strip()
862|        length = len(stripped)
863|
864|        # Light: very short or pure command (unless strong reasoner signals present)
865|        if length < 30 and not _CODE_SIGNALS.search(stripped) and not _STRONG_REASONER.search(stripped):
866|            return "light"
867|        if _LIGHT_COMMANDS.search(stripped):
868|            return "light"
869|        if _LIGHT_SIGNALS.search(stripped) and length < 200:
870|            has_code = _CODE_SIGNALS.search(stripped)
871|            if has_code:
872|                code_match = has_code.group()
873|                if code_match in (".py", ".js", ".ts", ".go", ".rs"):
874|                    has_code = False
875|            if not has_code and not _MODERATE_REASONER.search(stripped):
876|                return "light"
877|
878|        # Reasoner check
879|        strong_hits = _count_matches(_STRONG_REASONER, stripped)
880|        moderate_hits = _count_matches(_MODERATE_REASONER, stripped)
881|
882|        if strong_hits >= 1:
883|            return "reasoner"
884|        if moderate_hits >= 2:
885|            return "reasoner"
886|        if moderate_hits >= 1 and length > 400:
887|            return "reasoner"
888|        code_hits = _count_matches(_CODE_SIGNALS, stripped)
889|        if code_hits >= 5 and length > 300:
890|            return "reasoner"
891|
892|        # Code/engineering → pro (default)
893|        if code_hits >= 1 or length >= 30:
894|            return "pro"
895|
896|        return "pro"

1078|    def _build_fallback_chain(self) -> list[str]:
1079|        """动态构建 fallback 链：免费优先，付费兜底。
1080|
1081|        每个 provider 收集 pro + light 模型，按 cost_tier 排序。
1082|        """
1083|        chain: list[str] = []
1084|        try:
1085|            from core.provider import get_provider_manager
1086|            mgr = get_provider_manager()
1087|            seen: set[str] = set()
1088|            # Sort: free providers first, then by fallback priority
1089|            def _sort_key(pid):
1090|                p = mgr.providers.get(pid, {})
1091|                is_free = 0 if p.get("cost_tier") == "free" else 1
1092|                try: idx = mgr.fallback_priority.index(pid)
1093|                except ValueError: idx = 99
1094|                return (is_free, idx)
1095|            sorted_pids = sorted(mgr.fallback_priority, key=_sort_key)
1096|            for pid in sorted_pids:
1097|                provider = mgr.providers.get(pid, {})
1098|                models = provider.get("models", {})
1099|                for tier_key in ("pro", "light"):
1100|                    mid = models.get(tier_key, "")
1101|                    if mid and isinstance(mid, str) and mid not in seen:
1102|                        chain.append(mid); seen.add(mid)
1103|                for mid_val in models.values():
1104|                    if isinstance(mid_val, str) and mid_val not in seen:
1105|                        chain.append(mid_val); seen.add(mid_val)
1106|        except (ImportError, OSError, RuntimeError):
1107|            pass
1108|        return chain or ["deepseek-v4-pro", "deepseek-chat", "deepseek-v4-flash"]

1120|    def with_fallback(self, client, **chat_kwargs) -> dict:
1121|        """Call client.chat with automatic fallback on failure."""
1122|        model = chat_kwargs.get("model", self.primary)
1123|        last_error = None
1124|
1125|        while model:
1126|            try:
1127|                chat_kwargs["model"] = model
1128|                return client.chat(**chat_kwargs)
1129|            except (OSError, ValueError, RuntimeError) as e:
1130|                last_error = e
1131|                next_model = self.get_fallback(model)
1132|                if next_model and next_model != model:
1133|                    model = next_model
1134|                else:
1135|                    break
1136|
1137|        assert last_error is not None  # guaranteed by loop logic
1138|        raise last_error
