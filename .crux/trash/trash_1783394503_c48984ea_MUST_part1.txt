 += 1
        return msg

    def stream_chunk(self, text: str) -> Message | None:
        if not self._active_stream_id:
            return None
        msg = self._by_id[self._active_stream_id]
        msg.raw_text += text
        msg.updated_at = datetime.now()
        self._version += 1
        return msg

    def stream_end(self) -> Message | None:
        if not self._active_stream_id:
            return None
        msg = self._by_id[self._active_stream_id]
        msg.streaming = False
        msg.update_code_blocks()
        self._active_stream_id = None
        self._version += 1
        return msg

    def all(self) -> list[Message]:
        return list(self._messages)

    def get(self, message_id: str) -> Message | None:
        return self._by_id.get(message_id)

    def last_assistant(self) -> Message | None:
        for msg in reversed(self._messages):
            if msg.role == "assistant":
                return msg
        return None

class MessagePane:
    def __init__(self, message_store: MessageStore):
        self.message_store = message_store
        self._rendered_lines: list[tuple[str, str, str | None]] = []
        # role, text, message_id

    def append_message(self, role: str, text: str):
        msg = self.message_store.append(role, text)
        self._rebuild_lines()
        self.invalidate()

    def stream_start(self, role: str):
        self.message_store.stream_start(role)
        self._rebuild_lines()
        self.invalidate()

    def stream_chunk(self, text: str):
        self.message_store.stream_chunk(text)
        self._rebuild_lines()
        if self._pinned:
            self.scroll_to_bottom()
        self.invalidate()

    def stream_end(self):
        self.message_store.stream_end()
        self._rebuild_lines()
        if self._pinned:
            self.scroll_to_bottom()
        self.invalidate()

    def _rebuild_lines(self):
        self._rendered_lines.clear()
        for msg in self.message_store.all():
            for line in msg.raw_text.splitlines() or [""]:
                self._rendered_lines.append((msg.role, line, msg.id))
