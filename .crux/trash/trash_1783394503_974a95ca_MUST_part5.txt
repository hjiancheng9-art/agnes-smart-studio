@kb.add("f9")
def _(event):
    self.native_select = not self.native_select
    if self.native_select:
        self.input_router.set_mode(InputMode.NATIVE_SELECT)
        self._set_status("原生选择模式：TUI 暂停鼠标捕获，按 F9 返回")
    else:
        self.input_router.set_mode(InputMode.NORMAL)
        self._set_status("已恢复 TUI 鼠标控制")
    event.app.invalidate()
