结论先说：你现在的 TUI 已经不是“缺功能”，而是进入了 状态模型不够统一 的阶段。

核心问题有两个：

1. 消息仍然是屏幕行模型，而不是消息对象模型
2. 键盘/鼠标/复制/详情/运行控制没有统一 InputMode 路由

所以很多 bug 表面上是复制、详情页、Tab、↑↓冲突，本质上都来自这两点。

# ui/message_store.py
from dataclasses import dataclass, field
from datetime import datetime
from typing import Literal
import uuid
import re

Role = Literal["user", "assistant", "system", "tool"]

@dataclass
class CodeBlock:
    index: int
    language: str
    content: str
    start_line: int
    end_line: int

@dataclass
class Message:
    id: str
    role: Role
    raw_text: str = ""
    created_at: datetime = field(default_factory=datetime.now)
    updated_at: datetime = field(default_factory=datetime.now)
    streaming: bool = False
    code_blocks: list[CodeBlock] = field(default_factory=list)

    def update_code_blocks(self) -> None:
        self.code_blocks = extract_code_blocks(self.raw_text)

def extract_code_blocks(text: str) -> list[CodeBlock]:
    blocks: list[CodeBlock] = []
    pattern = re.compile(r"```(\w+)?\n(.*?)```", re.DOTALL)
    lines = text.splitlines()

    for i, match in enumerate(pattern.finditer(text), start=1):
        before = text[:match.start()]
        start_line = before.count("\n") + 1
        content = match.group(2)
        end_line = start_line + content.count("\n") + 1
        blocks.append(CodeBlock(
            index=i,
            language=match.group(1) or "",
            content=content,
            start_line=start_line,
            end_line=end_line,
        ))
    return blocks

class MessageStore:
    def __init__(self) -> None:
        self._messages: list[Message] = []
        self._by_id: dict[str, Message] = {}
        self._active_stream_id: str | None = None
        self._version = 0

    @property
    def version(self) -> int:
        return self._version

    def append(self, role: Role, text: str) -> Message:
        msg = Message(id=str(uuid.uuid4()), role=role, raw_text=text)
        msg.update_code_blocks()
        self._messages.append(msg)
        self._by_id[msg.id] = msg
        self._version += 1
        return msg

    def stream_start(self, role: Role) -> Message:
        msg = Message(id=str(uuid.uuid4()), role=role, streaming=True)
        self._messages.append(msg)
        self._by_id[msg.id] = msg
        self._active_stream_id = msg.id
        self._version += 1
        return msg

    def stream_chunk(self, text: str) -> Message | None:
        if not self._active_stream_id:
            return None
        msg = self._by_id[self._active_stream_id]
        msg.raw_text += text
        msg.updated_at = datetime.now()
        self._version += 1
        return msg

    def stream_end(self) -> Message | None:
        if not self._active_stream_id:
            return None
        msg = self._by_id[self._active_stream_id]
        msg.streaming = False
        msg.update_code_blocks()
        self._active_stream_id = None
        self._version += 1
        return msg

    def all(self) -> list[Message]:
        return list(self._messages)

    def get(self, message_id: str) -> Message | None:
        return self._by_id.get(message_id)

    def last_assistant(self) -> Message | None:
        for msg in reversed(self._messages):
            if msg.role == "assistant":
                return msg
        return None
