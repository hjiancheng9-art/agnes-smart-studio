# ui/copy_manager.py
class CopyManager:
    def __init__(self, message_store: MessageStore, clipboard):
        self.message_store = message_store
        self.clipboard = clipboard
        self.focused_message_id: str | None = None
        self.focused_code_block_index: int | None = None

    def get_focused_or_none(self):
        if self.focused_message_id:
            return self.message_store.get(self.focused_message_id)
        return None

    def copy_focused(self) -> str:
        msg = self.get_focused_or_none()
        if not msg:
            raise RuntimeError("NO_FOCUSED_MESSAGE")
        self.clipboard.copy(msg.raw_text)
        return msg.id

    def copy_last_assistant(self) -> str:
        msg = self.message_store.last_assistant()
        if not msg:
            raise RuntimeError("NO_ASSISTANT_MESSAGE")
        self.clipboard.copy(msg.raw_text)
        return msg.id
