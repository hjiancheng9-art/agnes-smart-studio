from enum import Enum

class InputMode(str, Enum):
    NORMAL = "normal"
    STREAMING = "streaming"
    DETAIL = "detail"
    COPY = "copy"
    NATIVE_SELECT = "native_select"
    COMMAND = "command"

class InputRouter:
    def __init__(self):
        self.mode = InputMode.NORMAL

    def set_mode(self, mode: InputMode):
        self.mode = mode

    def is_detail(self):
        return self.mode == InputMode.DETAIL
