t_code_block()
        self.message_pane.scroll_to_message_line(msg.id, block.start_line)
        self._set_status(f"代码块 {block.index}/{len(msg.code_blocks)} · 按 c 复制")
    except RuntimeError:
        self._set_status("当前消息没有代码块")

c 如果有 focused_code_block，则复制代码块；否则复制消息。
