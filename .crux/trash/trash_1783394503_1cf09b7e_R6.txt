def focus_next_code_block(self):
    msg = self.get_focused_or_none() or self.message_store.last_assistant()
    if not msg or not msg.code_blocks:
        raise RuntimeError("NO_CODE_BLOCK")

    if self.focused_message_id != msg.id:
        self.focused_message_id = msg.id
        self.focused_code_block_index = 1
    else:
        current = self.focused_code_block_index or 0
        self.focused_code_block_index = (current % len(msg.code_blocks)) + 1

    block = msg.code_blocks[self.focused_code_block_index - 1]
    return msg, block
