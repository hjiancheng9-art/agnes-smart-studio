    except Exception as e:
        print(f"TUI 启动失败: {e}", file=sys.stderr)
        print("回退到纯文本模式。")
        _chat_plain_session(session, cli, wire, session_id)

    # ── Cleanup ──
    if wire:
        with contextlib.suppress(OSError):
            wire.end_session()
