下面这套代码是按你的环境设计的：Windows 11 + Edge CDP 9222 + Playwright Python + 外层 run_python 30s 硬超时。

核心策略：

所有 Playwright 单动作 timeout 都小于 30s，避免卡死到外层硬超时。

点击不依赖默认导航等待，尤其是 ChatGPT 这种 SPA 页面，侧边栏链接常常不是传统导航。

locator.click / fill 失败后自动降级：普通动作 → force → JS click / DOM 注入 → 鼠标坐标兜底。

每次失败自动恢复页面：Escape、window.stop、domcontentloaded、必要时 reload。

定位优先级：role / testid / 可见文本 / CSS 兜底。

Playwright 官方说明里，Locator 是自动等待和重试的核心；click() 会等待元素可见、稳定、可接收事件、enabled 等 actionability checks，条件不满足会直到 timeout；fill() 会聚焦并触发 input 事件；现代页面不应盲等完整 load，更适合用 domcontentloaded 加应用级元素等待。

import re
import time
import traceback
from dataclasses import dataclass
from typing import Callable, Iterable, Optional, Any

from playwright.sync_api import (
    sync_playwright,
    Browser,
    BrowserContext,
    Page,
    Locator,
    TimeoutError as PlaywrightTimeoutError,
    Error as PlaywrightError,
)


@dataclass
class TimeoutProfile:
    # 必须显著小于 run_python 的 30s 硬超时
    connect_ms: int = 4_000
    default_action_ms: int = 5_000
    default_nav_ms: int = 8_000

    # 单个危险动作的短超时
    locator_wait_ms: int = 1_500
    click_ms: int = 2_500
    force_click_ms: int = 1_200
    js_click_ms: int = 1_000
    fill_ms: int = 3_000
    dom_fill_ms: int = 1_000

    # 页面恢复
    app_ready_ms: int = 5_000
    recover_ms: int = 4_000

    # 整个脚本预算，留给外层 run_python 退出
    total_budget_sec: float = 25.0


class Deadline:
    def __init__(self, seconds: float):
        self.start = time.monotonic()
        self.seconds = seconds

    def left_sec(self) -> float:
        return self.seconds - (time.monotonic() - self.start)

    def left_ms(self, desired_ms: int, min_ms: int = 300) -> int:
        left = int(max(0, self.left_sec() * 1000))
        # 给收尾留 800ms
        allowed = min(desired_ms, max(min_ms, left - 800))
        if allowed <= 0:
            raise TimeoutError("Global deadline exceeded before action")
        return allowed

    def ensure(self):
        if self.left_sec() <= 0:
            raise TimeoutError("Global deadline exceeded")


class EdgeChatGPTBot:
    CHATGPT_HOSTS = ("chatgpt.com", "chat.openai.com")

    COMPOSER_SELECTORS = [
        'textarea[data-testid="prompt-textarea"]',
        '#prompt-textarea',
        '[contenteditable="true"][data-testid="prompt-textarea"]',
        '[contenteditable="true"][role="textbox"]',
        'textarea',
        '[contenteditable="true"]',
    ]

    APP_READY_SELECTORS = [
        'main',
        'textarea[data-testid="prompt-textarea"]',
        '#prompt-textarea',
        '[contenteditable="true"][role="textbox"]',
        'nav',
        'aside',
    ]

    def __init__(
        self,
        cdp_url: str = "http://127.0.0.1:9222",
        timeouts: TimeoutProfile = TimeoutProfile(),
    ):
        self.cdp_url = cdp_url
        self.t = timeouts
        self.deadline = Deadline(self.t.total_budget_sec)

        self.pw = None
        self.browser: Optional[Browser] = None
        self.context: Optional[BrowserContext] = None
        self.page: Optional[Page] = None

    # -------------------------
    # lifecycle
    # -------------------------

    def connect(self) -> Page:
        self.deadline.ensure()

        self.pw = sync_playwright().start()
        self.browser = self.pw.chromium.connect_over_cdp(
            self.cdp_url,
            timeout=self.deadline.left_ms(self.t.connect_ms),
        )

        if self.browser.contexts:
            self.context = self.browser.contexts[0]
        else:
            self.context = self.browser.new_context()

        self.context.set_default_timeout(self.t.default_action_ms)
        self.context.set_default_navigation_timeout(self.t.default_nav_ms)

        self.page = self._pick_chatgpt_page() or self.context.new_page()
        self._configure_page(self.page)

        return self.page

    def close(self):
        # CDP 连接到已有 Edge，不建议 close browser，会关掉用户浏览器。
        try:
            if self.pw:
                self.pw.stop()
        except Exception:
            pass

    def _pick_chatgpt_page(self) -> Optional[Page]:
        if not self.context:
            return None

        for p in self.context.pages:
            if not p.is_closed() and any(host in p.url for host in self.CHATGPT_HOSTS):
                return p

        for p in self.context.pages:
            if not p.is_closed():
                return p

        return None

    def _configure_page(self, page: Page):
        page.set_default_timeout(self.t.default_action_ms)
        page.set_default_navigation_timeout(self.t.default_nav_ms)

        # 避免 dialog 卡住
        page.on("dialog", lambda d: d.dismiss())

    # -------------------------
    # page loading / readiness
    # -------------------------

    def goto_chatgpt(self, url: str = "https://chatgpt.com/") -> None:
        page = self.require_page()

        try:
            page.goto(
                url,
                wait_until="domcontentloaded",
                timeout=self.deadline.left_ms(self.t.default_nav_ms),
            )
        except PlaywrightTimeoutError:
            self.recover_page(reason="goto domcontentloaded timeout")

        self.wait_app_ready()

    def wait_app_ready(self) -> None:
        page = self.require_page()
        self.deadline.ensure()

        # 不盲等 networkidle。ChatGPT 这类页面可能长期有网络活动。
        try:
            page.wait_for_load_state(
                "domcontentloaded",
                timeout=self.deadline.left_ms(2_000),
            )
        except Exception:
            pass

        self.wait_for_any_visible(
            self.APP_READY_SELECTORS,
            timeout_ms=self.deadline.left_ms(self.t.app_ready_ms),
        )

    def wait_for_any_visible(
        self,
        selectors: Iterable[str],
        timeout_ms: int,
    ) -> Locator:
        page = self.require_page()
        deadline = time.monotonic() + timeout_ms / 1000
        last_error = None

        while time.monotonic() < deadline:
            for selector in selectors:
                try:
                    loc = page.locator(selector).first
                    # 兼容写法：Playwright 的 .first 是 property
                    if callable(loc):
                        loc = page.locator(selector).first()
                    loc.wait_for(state="visible", timeout=300)
                    return loc
                except Exception as e:
                    last_error = e

            time.sleep(0.08)

        raise PlaywrightTimeoutError(
            f"No visible selector found in {timeout_ms}ms. Last error: {last_error}"
        )

    # -------------------------
    # robust action wrapper
    # -------------------------

    def do(
        self,
        name: str,
        fn: Callable[[], Any],
        retries: int = 2,
        recover: bool = True,
    ) -> Any:
        last_error = None

        for attempt in range(retries + 1):
            self.deadline.ensure()

            try:
                return fn()

            except PlaywrightTimeoutError as e:
                last_error = e
                print(f"[timeout] {name}, attempt={attempt + 1}/{retries + 1}: {e}")
                if recover:
                    self.recover_page(reason=f"{name} timeout")

            except PlaywrightError as e:
                last_error = e
                print(f"[playwright-error] {name}, attempt={attempt + 1}/{retries + 1}: {e}")
                if recover:
                    self.recover_page(reason=f"{name} playwright error")

            except Exception as e:
                last_error = e
                print(f"[error] {name}, attempt={attempt + 1}/{retries + 1}: {e}")
                if recover:
                    self.recover_page(reason=f"{name} generic error")

        raise RuntimeError(f"Action failed after retries: {name}") from last_error

    def recover_page(self, reason: str = "") -> None:
        print(f"[recover] {reason}")

        page = self.page

        if page is None or page.is_closed():
            self.page = self._pick_chatgpt_page() or self.context.new_page()
            self._configure_page(self.page)
            return

        # 1. 尝试停止当前加载
        try:
            page.evaluate("window.stop()")
        except Exception:
            pass

        # 2. 关闭可能的浮层/菜单/拖拽状态
        try:
            page.keyboard.press("Escape", timeout=800)
        except Exception:
            pass

        # 3. 如果页面还能响应，只等应用级 ready
        try:
            self.wait_app_ready()
            return
        except Exception:
            pass

        # 4. 轻量 reload，仍然只等 domcontentloaded
        try:
            page.reload(
                wait_until="domcontentloaded",
                timeout=self.deadline.left_ms(self.t.recover_ms),
            )
            self.wait_app_ready()
            return
        except Exception:
            pass

        # 5. 最后兜底：开新 tab，不关闭用户原 tab
        try:
            self.page = self.context.new_page()
            self._configure_page(self.page)
            self.page.goto(
                "https://chatgpt.com/",
                wait_until="domcontentloaded",
                timeout=self.deadline.left_ms(self.t.default_nav_ms),
            )
            self.wait_app_ready()
        except Exception as e:
            raise RuntimeError(f"Failed to recover page: {e}") from e

    # -------------------------
    # locator best practice
    # -------------------------

    def require_page(self) -> Page:
        if self.page is None:
            raise RuntimeError("Page not initialized. Call connect() first.")
        if self.page.is_closed():
            self.page = self._pick_chatgpt_page() or self.context.new_page()
            self._configure_page(self.page)
        return self.page

    def by_sidebar_text(self, text: str) -> list[Locator]:
        """
        定位优先级：
        1. role=link/button + accessible name
        2. nav/aside 内文本
        3. 普通文本兜底
        """
        page = self.require_page()
        pattern = re.compile(re.escape(text), re.I)

        return [
            page.get_by_role("link", name=pattern).first,
            page.get_by_role("button", name=pattern).first,
            page.locator("nav a, aside a, nav button, aside button").filter(has_text=text).first,
            page.locator(f"text={text}").first,
        ]

    def first_usable(self, candidates: Iterable[Locator], timeout_ms: int = 1_500) -> Locator:
        last_error = None

        for loc in candidates:
            try:
                # 兼容 .first property / method
                if callable(loc):
                    loc = loc()
                loc.wait_for(state="attached", timeout=timeout_ms)
                return loc
            except Exception as e:
                last_error = e

        raise PlaywrightTimeoutError(f"No usable locator found. Last error: {last_error}")

    # -------------------------
    # robust click
    # -------------------------

    def safe_click(self, locator: Locator, name: str = "click") -> None:
        def _inner():
            self._safe_click_once(locator, name=name)

        return self.do(name, _inner, retries=2, recover=True)

    def _safe_click_once(self, locator: Locator, name: str = "click") -> None:
        page = self.require_page()
        old_url = page.url

        # 1. 等 attached/visible，时间短
        try:
            locator.wait_for(
                state="attached",
                timeout=self.deadline.left_ms(self.t.locator_wait_ms),
            )
        except Exception:
            pass

        try:
            locator.wait_for(
                state="visible",
                timeout=self.deadline.left_ms(self.t.locator_wait_ms),
            )
        except Exception:
            pass

        # 2. 滚动到可见区域
        try:
            locator.scroll_into_view_if_needed(
                timeout=self.deadline.left_ms(self.t.locator_wait_ms)
            )
        except Exception:
            pass

        # 3. 普通 click：禁用 after navigation 等待，避免 SPA/侧边栏卡住
        try:
            self._locator_click(locator, timeout_ms=self.t.click_ms, force=False, no_wait_after=True)
            self._after_action_wait(old_url)
            return
        except Exception as e1:
            print(f"[click-normal-failed] {name}: {e1}")

        # 4. force click：跳过部分 actionability 检查
        try:
            self._locator_click(locator, timeout_ms=self.t.force_click_ms, force=True, no_wait_after=True)
            self._after_action_wait(old_url)
            return
        except Exception as e2:
            print(f"[click-force-failed] {name}: {e2}")

        # 5. JS click：绕过 Playwright actionability
        try:
            locator.evaluate(
                """el => {
                    el.scrollIntoView({block: 'center', inline: 'center'});
                    el.click();
                }""",
                timeout=self.deadline.left_ms(self.t.js_click_ms),
            )
            self._after_action_wait(old_url)
            return
        except Exception as e3:
            print(f"[click-js-failed] {name}: {e3}")

        # 6. 坐标点击兜底
        try:
            box = locator.bounding_box(timeout=self.deadline.left_ms(1_000))
            if not box:
                raise RuntimeError("No bounding box")
            page.mouse.click(
                box["x"] + box["width"] / 2,
                box["y"] + box["height"] / 2,
            )
            self._after_action_wait(old_url)
            return
        except Exception as e4:
            raise RuntimeError(f"All click strategies failed for {name}: {e4}") from e4

    def _locator_click(
        self,
        locator: Locator,
        timeout_ms: int,
        force: bool,
        no_wait_after: bool,
    ) -> None:
        kwargs = {
            "timeout": self.deadline.left_ms(timeout_ms),
            "force": force,
        }

        # 某些版本有 no_wait_after，某些版本可能弃用但仍兼容。
        try:
            locator.click(no_wait_after=no_wait_after, **kwargs)
        except TypeError:
            locator.click(**kwargs)

    def _after_action_wait(self, old_url: str) -> None:
        page = self.require_page()

        # 不等 networkidle。
        # 如果 URL 变化，只等 domcontentloaded；如果没变化，只等应用 ready。
        try:
            if page.url != old_url:
                page.wait_for_load_state(
                    "domcontentloaded",
                    timeout=self.deadline.left_ms(2_000),
                )
        except Exception:
            pass

        try:
            self.wait_app_ready()
        except Exception:
            # 点击侧边栏后主区域可能正在切 conversation，不强行失败
            pass

    def click_sidebar_item(self, text: str) -> None:
        def _inner():
            loc = self.first_usable(
                self.by_sidebar_text(text),
                timeout_ms=self.deadline.left_ms(self.t.locator_wait_ms),
            )
            self._safe_click_once(loc, name=f"sidebar:{text}")

        return self.do(f"click_sidebar_item:{text}", _inner, retries=2, recover=True)

    # -------------------------
    # robust fill / input
    # -------------------------

    def find_composer(self) -> Locator:
        return self.wait_for_any_visible(
            self.COMPOSER_SELECTORS,
            timeout_ms=self.deadline.left_ms(self.t.app_ready_ms),
        )

    def safe_fill_composer(self, text: str) -> None:
        def _inner():
            composer = self.find_composer()
            self._safe_fill_once(composer, text)

        return self.do("safe_fill_composer", _inner, retries=2, recover=True)

    def _safe_fill_once(self, locator: Locator, text: str) -> None:
        page = self.require_page()

        # 1. 聚焦
        try:
            locator.click(timeout=self.deadline.left_ms(1_000), force=True)
        except Exception:
            try:
                locator.focus(timeout=self.deadline.left_ms(800))
            except Exception:
                pass

        # 2. Playwright fill：适合 textarea/input/contenteditable
        try:
            locator.fill(
                text,
                timeout=self.deadline.left_ms(self.t.fill_ms),
            )
            return
        except Exception as e1:
            print(f"[fill-normal-failed] {e1}")

        # 3. DOM 注入：适合 ChatGPT 输入框卡顿、React 状态没同步时
        try:
            locator.evaluate(
                """
                (el, value) => {
                    el.focus();

                    const fire = (type, init = {}) => {
                        el.dispatchEvent(new InputEvent(type, {
                            bubbles: true,
                            cancelable: true,
                            composed: true,
                            inputType: init.inputType || 'insertText',
                            data: init.data || value
                        }));
                    };

                    if (el.isContentEditable) {
                        const selection = window.getSelection();
                        const range = document.createRange();

                        range.selectNodeContents(el);
                        selection.removeAllRanges();
                        selection.addRange(range);

                        document.execCommand('delete', false, null);
                        document.execCommand('insertText', false, value);

                        fire('input', {inputType: 'insertText', data: value});
                        el.dispatchEvent(new Event('change', {bubbles: true}));
                        return true;
                    }

                    if ('value' in el) {
                        const proto =
                            el.tagName === 'TEXTAREA'
                                ? HTMLTextAreaElement.prototype
                                : HTMLInputElement.prototype;

                        const desc = Object.getOwnPropertyDescriptor(proto, 'value');
                        if (desc && desc.set) {
                            desc.set.call(el, value);
                        } else {
                            el.value = value;
                        }

                        fire('input', {inputType: 'insertText', data: value});
                        el.dispatchEvent(new Event('change', {bubbles: true}));
                        return true;
                    }

                    el.textContent = value;
                    fire('input', {inputType: 'insertText', data: value});
                    el.dispatchEvent(new Event('change', {bubbles: true}));
                    return true;
                }
                """,
                text,
                timeout=self.deadline.left_ms(self.t.dom_fill_ms),
            )
            return
        except Exception as e2:
            print(f"[fill-dom-failed] {e2}")

        # 4. 键盘兜底。insert_text 通常比 type 快。
        try:
            locator.click(timeout=self.deadline.left_ms(1_000), force=True)
            page.keyboard.press("Control+A", timeout=800)
            page.keyboard.press("Backspace", timeout=800)
            page.keyboard.insert_text(text)
            return
        except Exception as e3:
            raise RuntimeError(f"All fill strategies failed: {e3}") from e3

    def submit_composer(self) -> None:
        page = self.require_page()

        def _inner():
            # 优先点发送按钮
            candidates = [
                page.locator('[data-testid="send-button"]').first,
                page.get_by_role("button", name=re.compile("Send|发送|提交", re.I)).first,
                page.locator('button[aria-label*="Send"], button[aria-label*="发送"]').first,
            ]

            try:
                loc = self.first_usable(candidates, timeout_ms=1_000)
                self._safe_click_once(loc, name="send-button")
                return
            except Exception:
                pass

            # 兜底 Enter
            page.keyboard.press("Enter", timeout=self.deadline.left_ms(1_000))

        return self.do("submit_composer", _inner, retries=1, recover=True)

    # -------------------------
    # high level example
    # -------------------------

    def ask(self, text: str) -> None:
        self.safe_fill_composer(text)
        self.submit_composer()


def main():
    bot = EdgeChatGPTBot(
        cdp_url="http://127.0.0.1:9222",
        timeouts=TimeoutProfile(
            connect_ms=4_000,
            default_action_ms=5_000,
            default_nav_ms=8_000,
            locator_wait_ms=1_500,
            click_ms=2_500,
            force_click_ms=1_200,
            js_click_ms=1_000,
            fill_ms=3_000,
            dom_fill_ms=1_000,
            app_ready_ms=5_000,
            recover_ms=4_000,
            total_budget_sec=25.0,
        ),
    )

    try:
        bot.connect()
        bot.goto_chatgpt()

        # 示例 1：点击侧边栏会话
        # bot.click_sidebar_item("你的会话标题")

        # 示例 2：输入并发送
        bot.ask("你好，请用一句话回复我。")

    except Exception:
        traceback.print_exc()
    finally:
        bot.close()


if __name__ == "__main__":
    main()

侧边栏点击卡住时，本质上多半不是“点不到”，而是 Playwright 在等 actionability 或 navigation。上面的方案会把点击拆成：

Python
运行
normal click
→ force click
→ JS el.click()
→ mouse coordinate click
→ recover_page()
→ retry

这样就不会让一次 locator.click() 直接拖到你的 30s 硬超时。
