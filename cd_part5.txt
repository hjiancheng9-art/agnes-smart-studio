def test_parameter_policy_clamps_bad_values():

    from comfyflow.task_spec import TaskSpec
    from comfyflow.runtime_budget import RuntimeBudget

    task = TaskSpec(
        intent="txt2img",
        prompt="cat",
        width=9999,
        height=1001,
        steps=100,
        cfg=99,
        batch_size=8,
        aspect_ratio="9:16",
    )

    budget = RuntimeBudget(
        tier="low_6gb",
        vram_gb=6,
        max_width=768,
        max_height=1152,
        max_batch_size=1,
        allow_sdxl=False,
        allow_flux=False,
        allow_video=False,
    )

    spec = ParameterPolicy().apply(task, budget)

    assert spec.width % 8 == 0
    assert spec.height % 8 == 0
    assert spec.width <= 768
    assert spec.height <= 1152
    assert 8 <= spec.steps <= 40
    assert 1 <= spec.cfg <= 15
