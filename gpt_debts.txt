# comfyflow/safety_gate.py
from dataclasses import dataclass, field
from pathlib import Path
from typing import Any


@dataclass
class SafetyIssue:
    code: str
    message: str
    severity: str = "error"


@dataclass
class SafetyReport:
    passed: bool
    issues: list[SafetyIssue] = field(default_factory=list)


class WorkflowSafetyGate:
    def __init__(
        self,
        allowed_class_types: set[str],
        blocked_class_keywords: list[str] | None = None,
    ):
        self.allowed_class_types = allowed_class_types
        self.blocked_class_keywords = blocked_class_keywords or [
            "exec",
            "subprocess",
            "shell",
            "cmd",
            "powershell",
            "download",
            "http",
            "request",
        ]

    def check(self, workflow: dict[str, Any]) -> SafetyReport:
        issues: list[SafetyIssue] = []

        if not isinstance(workflow, dict):
            issues.append(SafetyIssue("bad_workflow_type", "workflow 必须是 dict"))
            return SafetyReport(False, issues)

        for node_id, node in workflow.items():
            if not isinstance(node_id, str):
                issues.append(SafetyIssue("bad_node_id", f"node_id 必须是字符串: {node_id}"))

            class_type = node.get("class_type")
            if not class_type:
                issues.append(SafetyIssue("missing_class_type", f"节点 {node_id} 缺少 class_type"))
                continue

            if class_type not in self.allowed_class_types:
                issues.append(SafetyIssue(
                    "unknown_class_type",
                    f"节点 {node_id} 使用了未注册节点: {class_type}",
                ))

            low = class_type.lower()
            if any(k in low for k in self.blocked_class_keywords):
                issues.append(SafetyIssue(
                    "risky_node_type",
                    f"节点 {node_id} class_type 看起来有风险: {class_type}",
                ))

        return SafetyReport(
            passed=not any(i.severity == "error" for i in issues),
            issues=issues,
        )

def safe_queue_prompt(client, workflow, object_info):
    gate = WorkflowSafetyGate(allowed_class_types=set(object_info.keys()))
    report = gate.check(workflow)

    if not report.passed:
        return {
            "success": False,
            "stage": "safety_gate",
            "errors": [i.__dict__ for i in report.issues],

    return client.queue_prompt(workflow)

# comfyflow/safe_subprocess.py
import subprocess
from dataclasses import dataclass


@dataclass
class CommandResult:
    ok: bool
    stdout: str
    stderr: str
    returncode: int


ALLOWED_COMMANDS = {
    "nvidia-smi",
}


def run_safe_command(args: list[str], timeout: int = 10) -> CommandResult:
    if not args:
        raise ValueError("empty command")

    exe = args[0]

    if exe not in ALLOWED_COMMANDS:
        raise PermissionError(f"command not allowed: {exe}")

    completed = subprocess.run(
        args,
        shell=False,
        capture_output=True,
        text=True,
        timeout=timeout,
        check=False,
        encoding="utf-8",
        errors="ignore",
    )

    return CommandResult(
        ok=completed.returncode == 0,
        stdout=completed.stdout,
        stderr=completed.stderr,
        returncode=completed.returncode,
    )

from pathlib import Path


def resolve_under_root(root: str | Path, target: str | Path) -> Path:
    root_path = Path(root).expanduser().resolve()
    target_path = Path(target).expanduser().resolve()

    if root_path not in target_path.parents and target_path != root_path:
        raise PermissionError(f"path escapes root: {target_path}")

    return target_path

# comfyflow/node_id_allocator.py

class NodeIdAllocator:
    def __init__(self, start: int = 1):
        self._next = start

    def next(self) -> str:
        node_id = str(self._next)
        self._next += 1
        return node_id

    def reserve(self, node_id: int | str) -> str:
        node_id_str = str(node_id)
        self._next = max(self._next, int(node_id_str) + 1)
        return node_id_str

# comfyflow/manifest.py
import hashlib
import json
from dataclasses import dataclass, field
from datetime import datetime, timezone
from typing import Any


COMPILER_VERSION = "0.3.1"
PARAMETER_TABLE_VERSION = "2026-07-09"


def stable_json_dumps(data: dict[str, Any]) -> str:
    return json.dumps(
        data,
        ensure_ascii=False,
        sort_keys=True,
        separators=(",", ":"),
    )


def workflow_checksum(workflow: dict[str, Any]) -> str:
    raw = stable_json_dumps(workflow).encode("utf-8")
    return hashlib.sha256(raw).hexdigest()


@dataclass
class WorkflowManifest:
    compiler_version: str
    blueprint_id: str
    blueprint_version: str
    parameter_table_version: str
    workflow_checksum: str
    created_at: str
    task_intent: str
    quality_mode: str
    model_family: str
    hardware_tier: str
    notes: list[str] = field(default_factory=list)


def build_manifest(
    workflow: dict[str, Any],
    task,
    blueprint,
    hardware_budget,
) -> WorkflowManifest:
    return WorkflowManifest(
        compiler_version=COMPILER_VERSION,
        blueprint_id=blueprint.id,
        blueprint_version=getattr(blueprint, "version", "0.1.0"),
        parameter_table_version=PARAMETER_TABLE_VERSION,
        workflow_checksum=workflow_checksum(workflow),
        created_at=datetime.now(timezone.utc).isoformat(),
        task_intent=task.intent,
        quality_mode=task.quality_mode,
        model_family=task.model_family,
        hardware_tier=hardware_budget.tier,
    )

def choose_variant(user_or_task_id: str, experiment_name: str, variants: list[str]) -> str:
