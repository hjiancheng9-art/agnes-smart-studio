结论

Showrunner 主导，ComfyUI Agent 被调用。

关系应该是：

用户目标
  ↓
Showrunner：创意导演 / 制片人 / 总编排
  ↓
SkillStep / ShotSpec / CharacterSpec / AssetSpec
  ↓
ComfyUI Agent V4：摄影师 + 剪辑师 + 工作流执行器
  ↓
FlowCompiler → diagnose → repair → validate → execute
  ↓
图片 / 视频 / 中间资产 / 最终影片

不要让 ComfyUI Agent 反向调用 Showrunner。
Showrunner 管“做什么、为什么、顺序、风格一致性”；ComfyUI Agent 管“怎么生成、怎么修、怎么跑 ComfyUI workflow”。

# showrunner/comfy_adapter.py

from .contracts import ComfyJobSpec, ComfyJobResult, AssetRef
from comfyui_agent.agent import ComfyUIAgent
from comfyui_agent.flow_compiler import FlowCompiler


class ComfyUIAgentAdapter:
    def __init__(
        self,
        comfy_agent: ComfyUIAgent,
        flow_compiler: FlowCompiler,
    ):
        self.agent = comfy_agent
        self.compiler = flow_compiler

    def run_job(self, spec: ComfyJobSpec) -> ComfyJobResult:
        try:
            workflow = self.compiler.compile(
                task_type=spec.task_type,
                prompt=spec.prompt,
                negative_prompt=spec.negative_prompt,
                input_assets=spec.input_assets,
                actors=spec.actors,
                constraints=spec.constraints,
                workflow_hint=spec.workflow_hint,
            )

            diag = self.agent.diagnose(workflow)
            if not diag.ok:
                workflow = self.agent.repair(workflow, diag)

            validation = self.agent.validate(workflow)
            if not validation.ok:
                workflow = self.agent.repair(workflow, validation)

            result = self.agent.execute(workflow)

            return ComfyJobResult(
                ok=result.ok,
                job_id=spec.job_id,
                assets=[
                    AssetRef(
                        id=a.id,
                        kind=a.kind,
                        path=a.path,
                        meta=getattr(a, "meta", {}),
                    )
                    for a in result.assets
                ],
                workflow_json=workflow,
                diagnostics=[
                    getattr(diag, "to_dict", lambda: diag.__dict__)(),
                    getattr(validation, "to_dict", lambda: validation.__dict__)(),
                ],
                error=result.error if not result.ok else None,
            )

        except Exception as e:
            return ComfyJobResult(
                ok=False,
                job_id=spec.job_id,
                error=f"ComfyUI job failed: {type(e).__name__}: {e}",
            )

# showrunner/skill_step_mapper.py

from .contracts import SkillStep, ShowrunnerPlan, ComfyJobSpec, ActorCraftSpec


class SkillStepMapper:
    def to_comfy_job(
        self,
        step: SkillStep,
        plan: ShowrunnerPlan,
        runtime_assets: dict[str, list],
    ) -> ComfyJobSpec | None:
        if step.kind in {"script", "final_delivery"}:
            return None

        actors = [
            plan.actors[actor_id]
            for actor_id in step.actor_refs
            if actor_id in plan.actors
        ]

        input_assets = []
        for dep_id in step.depends_on:
            input_assets.extend(runtime_assets.get(dep_id, []))

        prompt = self._build_prompt(step, plan, actors)

        task_type = self._map_task_type(step.kind)

        return ComfyJobSpec(
            job_id=step.id,
            task_type=task_type,
            prompt=prompt,
            negative_prompt=self._build_negative_prompt(step, plan),
            actors=actors,
            input_assets=input_assets + step.input_assets,
            constraints={
                **plan.style_bible,
                **step.constraints,
            },
            workflow_hint=step.constraints.get("workflow_hint"),
        )

    def _map_task_type(self, kind: str) -> str:
        mapping = {
            "character_design": "txt2img",
            "scene_design": "txt2img",
            "storyboard": "txt2img",
            "txt2img": "txt2img",
            "img2img": "img2img",
            "image_to_video": "image_to_video",
            "video_extend": "video_extend",
            "upscale": "upscale",
            "edit": "edit",
        }

        if kind not in mapping:
            raise ValueError(f"Unsupported ComfyUI step kind: {kind}")

        return mapping[kind]

    def _build_prompt(
        self,
        step: SkillStep,
        plan: ShowrunnerPlan,
        actors: list[ActorCraftSpec],
    ) -> str:
        actor_block = "\n".join(
            [
                f"""
Character: {a.name}
Role: {a.role}
Core identity: {a.layer_1_core_identity}
Silhouette: {a.layer_2_visual_silhouette}
Costume/material: {a.layer_3_costume_material}
Motion: {a.layer_4_motion_signature}
Aura: {a.layer_5_emotional_aura}
"""
                for a in actors
            ]
        )

        style_block = "\n".join(
            f"{k}: {v}" for k, v in plan.style_bible.items()
        )

        return f"""
TASK:
{step.title}

PROMPT:
{step.prompt}

STYLE BIBLE:
{style_block}

ACTOR CRAFT:
{actor_block}

CONSTRAINTS:
{step.constraints}
""".strip()

    def _build_negative_prompt(
        self,
        step: SkillStep,
        plan: ShowrunnerPlan,
    ) -> str:
        negatives = []

        for actor_id in step.actor_refs:
            actor = plan.actors.get(actor_id)
            if actor:
                negatives.extend(actor.negative_traits)

        negatives.extend(step.constraints.get("negative_prompt", []))
        negatives.extend(plan.style_bible.get("global_negative_prompt", []))

        return ", ".join(dict.fromkeys(negatives))

Python

# showrunner/orchestrator.py

from .contracts import ShowrunnerPlan, ComfyJobResult
from .skill_step_mapper import SkillStepMapper
from .comfy_port import ComfyExecutorPort


class ShowrunnerOrchestrator:
    def __init__(
        self,
        comfy_executor: ComfyExecutorPort,
        mapper: SkillStepMapper,
    ):
        self.comfy = comfy_executor
        self.mapper = mapper

    def run(self, plan: ShowrunnerPlan) -> dict:
        runtime_assets: dict[str, list] = {}
        step_results: dict[str, ComfyJobResult | dict] = {}

        for step in plan.steps:
            self._ensure_dependencies_done(step, step_results)

            if step.kind == "script":
                step_results[step.id] = self._run_script_step(step, plan)
                continue

            if step.kind == "final_delivery":
                step_results[step.id] = self._run_delivery_step(
                    step=step,
                    plan=plan,
                    runtime_assets=runtime_assets,
                )
                continue

            comfy_job = self.mapper.to_comfy_job(
                step=step,
                plan=plan,
                runtime_assets=runtime_assets,
            )

            if comfy_job is None:
                continue

            result = self.comfy.run_job(comfy_job)
            step_results[step.id] = result

            if not result.ok:
                raise RuntimeError(
                    f"Step failed: {step.id} / {step.title}: {result.error}"
                )

            runtime_assets[step.id] = result.assets

        return {
            "ok": True,
            "goal": plan.goal,
            "assets": runtime_assets,
            "steps": step_results,
        }

    def _ensure_dependencies_done(self, step, step_results):
        missing = [
            dep for dep in step.depends_on
            if dep not in step_results
        ]

        if missing:
            raise RuntimeError(
                f"Step {step.id} has missing dependencies: {missing}"
            )

    def _run_script_step(self, step, plan):
        return {
            "ok": True,
            "kind": "script",
            "text": step.prompt,
        }

    def _run_delivery_step(self, step, plan, runtime_assets):
        return {
            "ok": True,
            "kind": "final_delivery",
            "assets": runtime_assets,
        }

# comfyui_agent/flow_compiler.py

class FlowCompiler:
    def __init__(self, node_registry, template_registry, validator):
        self.node_registry = node_registry
        self.template_registry = template_registry
        self.validator = validator

    def compile(
        self,
        task_type: str,
        prompt: str,
        negative_prompt: str = "",
        input_assets=None,
        actors=None,
        constraints=None,
        workflow_hint: str | None = None,
    ) -> dict:
        constraints = constraints or {}
        input_assets = input_assets or []
        actors = actors or []

        template = self.template_registry.select(
            task_type=task_type,
            hint=workflow_hint,
            constraints=constraints,
            input_assets=input_assets,
        )

        workflow = template.render(
            prompt=prompt,
            negative_prompt=negative_prompt,
            input_assets=input_assets,
            actors=actors,
            constraints=constraints,
        )

        self.node_registry.assert_no_hallucinated_nodes(workflow)
        self.validator.assert_valid(workflow)

        return workflow
8. Actor-craft 放哪一层？

Actor-craft 放在 Showrunner 层。

原因：

Actor-craft = 角色设计法 / 角色圣经 / 连续性约束
ComfyUI Agent = 具体执行 / 节点 / 模型 / workflow

所以 Actor-craft 不应该塞进 ComfyUI Agent 的核心逻辑。
它应该作为 Showrunner 生成的 ActorCraftSpec，在每个相关 SkillStep 里传给 ComfyUI Agent。

from showrunner.orchestrator import ShowrunnerOrchestrator
from showrunner.skill_step_mapper import SkillStepMapper
from showrunner.comfy_adapter import ComfyUIAgentAdapter

from comfyui_agent.agent import ComfyUIAgent
from comfyui_agent.flow_compiler import FlowCompiler
from comfyui_agent.node_registry import NodeRegistry
from comfyui_agent.workflow_validator import WorkflowValidator
from comfyui_agent.template_registry import TemplateRegistry


def build_system():
    node_registry = NodeRegistry()
    validator = WorkflowValidator(node_registry=node_registry)
    template_registry = TemplateRegistry()

    flow_compiler = FlowCompiler(
        node_registry=node_registry,
        template_registry=template_registry,
        validator=validator,
    )

    comfy_agent = ComfyUIAgent(
        flow_compiler=flow_compiler,
        validator=validator,
    )

    comfy_adapter = ComfyUIAgentAdapter(
        comfy_agent=comfy_agent,
        flow_compiler=flow_compiler,
    )

    showrunner = ShowrunnerOrchestrator(
        comfy_executor=comfy_adapter,
        mapper=SkillStepMapper(),
    )

    return showrunner
11. 一句话架构原则
Showrunner 负责编排创意意图，ComfyUI Agent 负责编译和执行工作流；Actor-craft 是上层角色圣经，通过契约下发，不侵入节点执行层。

最小落地优先级：

P0：包命名隔离，解决两个 agent.py 冲突
P1：定义 contracts.py，固定 SkillStep → ComfyJobSpec → ComfyJobResult
P2：写 ComfyUIAgentAdapter，让 Showrunner 能调用 ComfyUI Agent V4
P3：写 SkillStepMapper，把创意步骤映射成 Comfy 任务
P4：Actor-craft 上移到 Showrunner，作为角色连续性约束传入
P5：Showrunner_pipeline.py 改成只调 ShowrunnerOrchestrator，不直接碰 ComfyUI workflow
