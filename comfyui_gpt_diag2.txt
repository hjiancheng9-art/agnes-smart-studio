# comfyui_diagnose.py
# Python 3.10+
from __future__ import annotations

import json
import os
import sys
import time
import uuid
import socket
import traceback
from dataclasses import dataclass, asdict
from pathlib import Path
from typing import Any, Dict, List, Optional, Set, Tuple
from urllib.parse import urljoin

try:
    import requests
except ImportError:
    requests = None


Severity = str  # "info" | "warning" | "error" | "critical"


@dataclass
class DiagnoseItem:
    ok: bool
    severity: Severity
    evidence: Dict[str, Any]
    suggestion: str

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)


class ComfyUIDiagnoser:
    """
    diagnose pipeline:
    env_check -> model_check -> workflow_check -> prompt_check -> comfyui_check -> execution_check
    """

    MODEL_INPUT_KEYS = {
        "ckpt_name": "checkpoints",
        "vae_name": "vae",
        "lora_name": "loras",
        "control_net_name": "controlnet",
        "controlnet_name": "controlnet",
        "clip_name": "clip",
        "clip_name1": "clip",
        "clip_name2": "clip",
        "clip_name3": "clip",
        "clip_vision_name": "clip_vision",
        "unet_name": "diffusion_models",
        "diffusion_model_name": "diffusion_models",
        "text_encoder_name": "text_encoders",
        "upscale_model_name": "upscale_models",
    }

    DEFAULT_MODEL_DIRS = {
        "checkpoints": ["models/checkpoints"],
        "vae": ["models/vae"],
        "loras": ["models/loras"],
        "controlnet": ["models/controlnet", "models/controlnet_annotators"],
        "clip": ["models/clip"],
        "clip_vision": ["models/clip_vision"],
        "diffusion_models": ["models/diffusion_models", "models/unet"],
        "text_encoders": ["models/text_encoders"],
        "upscale_models": ["models/upscale_models"],
    }

    def __init__(
        self,
        comfyui_root: str | Path,
        comfyui_url: str = "http://127.0.0.1:8188",
        model_dirs: Optional[Dict[str, List[str]]] = None,
        timeout: float = 3.0,
        execute_probe: bool = False,
    ) -> None:
        self.comfyui_root = Path(comfyui_root).expanduser().resolve()
        self.comfyui_url = comfyui_url.rstrip("/") + "/"
        self.timeout = timeout
        self.execute_probe = execute_probe

        merged_dirs = dict(self.DEFAULT_MODEL_DIRS)
        if model_dirs:
            for k, v in model_dirs.items():
                merged_dirs[k] = v
        self.model_dirs = merged_dirs

    # ============================================================
    # Public API
    # ============================================================

    def diagnose(
        self,
        workflow: Optional[Dict[str, Any] | str | Path] = None,
        prompt_text: Optional[str] = None,
    ) -> Dict[str, Dict[str, Any]]:
        workflow_obj = self._load_workflow(workflow) if workflow is not None else None

        results: Dict[str, DiagnoseItem] = {}

        results["env_check"] = self.env_check()
        results["model_check"] = self.model_check(workflow_obj)
        results["workflow_check"] = self.workflow_check(workflow_obj)
        results["prompt_check"] = self.prompt_check(prompt_text, workflow_obj)
        results["comfyui_check"] = self.comfyui_check()
        results["execution_check"] = self.execution_check(workflow_obj, results)

        return {k: v.to_dict() for k, v in results.items()}

    # ============================================================
    # 1. env_check
    # ============================================================

    def env_check(self) -> DiagnoseItem:
        evidence: Dict[str, Any] = {
            "python": sys.version,
            "platform": sys.platform,
            "comfyui_root": str(self.comfyui_root),
            "requests_installed": requests is not None,
            "paths": {},
        }

        required_paths = {
            "root": self.comfyui_root,
            "models": self.comfyui_root / "models",
            "output": self.comfyui_root / "output",
            "custom_nodes": self.comfyui_root / "custom_nodes",
        }

        ok = True
        missing = []

        for name, path in required_paths.items():
            exists = path.exists()
            is_dir = path.is_dir()
            writable = self._is_writable_dir(path) if exists and is_dir else False
            evidence["paths"][name] = {
                "path": str(path),
                "exists": exists,
                "is_dir": is_dir,
                "writable": writable,
            }

            if name in {"root", "models"} and not exists:
                ok = False
                missing.append(str(path))

        if requests is None:
            ok = False

        if ok:
            return DiagnoseItem(
                ok=True,
                severity="info",
                evidence=evidence,
                suggestion="环境基础检查通过。",
            )

        return DiagnoseItem(
            ok=False,
            severity="critical" if not self.comfyui_root.exists() else "error",
            evidence={**evidence, "missing": missing},
            suggestion=(
                "请确认 comfyui_root 指向 ComfyUI 根目录，并安装 requests："
                "pip install requests。若 output/custom_nodes 不存在，可手动创建。"
            ),
        )

    # ============================================================
    # 2. model_check
    # ============================================================

    def model_check(self, workflow: Optional[Dict[str, Any]]) -> DiagnoseItem:
        evidence: Dict[str, Any] = {
            "comfyui_root": str(self.comfyui_root),
            "required_models": [],
            "missing_models": [],
            "search_dirs": {},
        }

        for model_type, dirs in self.model_dirs.items():
            evidence["search_dirs"][model_type] = [
                str((self.comfyui_root / d).resolve()) for d in dirs
            ]

        if workflow is None:
            return DiagnoseItem(
                ok=True,
                severity="warning",
                evidence=evidence,
                suggestion="未提供 workflow，跳过模型文件精确检查。建议传入 workflow JSON。",
            )

        required = self._extract_required_models(workflow)
        evidence["required_models"] = required

        missing = []
        for item in required:
            model_type = item["model_type"]
            filename = item["filename"]

            found = self._find_model_file(model_type, filename)
            item["found"] = found is not None
            item["resolved_path"] = str(found) if found else None

            if found is None:
                missing.append(item)

        evidence["missing_models"] = missing

        if missing:
            return DiagnoseItem(
                ok=False,
                severity="error",
                evidence=evidence,
                suggestion=(
                    "发现 workflow 引用的模型文件缺失。请把缺失文件放入对应 models 子目录，"
                    "或修改 workflow 中的模型名称。重点检查 checkpoints、vae、loras、controlnet、clip、text_encoders。"
                ),
            )

        return DiagnoseItem(
            ok=True,
            severity="info",
            evidence=evidence,
            suggestion="workflow 引用的模型文件均已找到。",
        )

    # ============================================================
    # 3. workflow_check
    # ============================================================

    def workflow_check(self, workflow: Optional[Dict[str, Any]]) -> DiagnoseItem:
        evidence: Dict[str, Any] = {
            "format": None,
            "node_count": 0,
            "edge_count": 0,
            "errors": [],
            "warnings": [],
        }

        if workflow is None:
            return DiagnoseItem(
                ok=False,
                severity="error",
                evidence=evidence,
                suggestion="未提供 workflow JSON，无法检查拓扑合法性。",
            )

        try:
            api_prompt = self._normalize_workflow_to_api_prompt(workflow)
            evidence["format"] = "api_prompt"
            evidence["node_count"] = len(api_prompt)
        except Exception as e:
            evidence["errors"].append(str(e))
            return DiagnoseItem(
                ok=False,
                severity="error",
                evidence=evidence,
                suggestion="workflow JSON 格式无法识别。请传入 ComfyUI API prompt 格式，或标准 UI workflow 格式。",
            )

        errors: List[str] = []
        warnings: List[str] = []
        graph: Dict[str, Set[str]] = {node_id: set() for node_id in api_prompt}
        reverse_graph: Dict[str, Set[str]] = {node_id: set() for node_id in api_prompt}
        edge_count = 0

        for node_id, node in api_prompt.items():
            if not isinstance(node, dict):
                errors.append(f"node {node_id} 不是 object")
                continue

            class_type = node.get("class_type")
            inputs = node.get("inputs")

            if not class_type or not isinstance(class_type, str):
                errors.append(f"node {node_id} 缺少 class_type")
            if not isinstance(inputs, dict):
                errors.append(f"node {node_id} 缺少 inputs object")
                continue

            for input_name, value in inputs.items():
                if self._is_link(value):
                    src_id, output_index = str(value[0]), value[1]
                    edge_count += 1

                    if src_id not in api_prompt:
                        errors.append(
                            f"node {node_id}.{input_name} 引用了不存在的上游节点 {src_id}"
                        )
                        continue

                    if not isinstance(output_index, int) or output_index < 0:
                        errors.append(
                            f"node {node_id}.{input_name} 的 output index 非法: {output_index}"
                        )
                        continue

                    graph[src_id].add(node_id)
                    reverse_graph[node_id].add(src_id)

        evidence["edge_count"] = edge_count

        cycle = self._detect_cycle(graph)
        if cycle:
            errors.append("workflow 拓扑存在环: " + " -> ".join(cycle))

        orphan_nodes = [
            node_id
            for node_id, parents in reverse_graph.items()
            if not parents and not self._looks_like_source_node(api_prompt[node_id])
        ]
        if orphan_nodes:
            warnings.append(f"存在无上游输入的非源节点: {orphan_nodes}")

        output_nodes = [
            node_id
            for node_id, children in graph.items()
            if not children
        ]
        if not output_nodes:
            errors.append("workflow 没有终端输出节点")
        else:
            evidence["terminal_nodes"] = output_nodes

        evidence["errors"] = errors
        evidence["warnings"] = warnings

        if errors:
            return DiagnoseItem(
                ok=False,
                severity="error",
                evidence=evidence,
                suggestion=(
                    "workflow JSON 拓扑非法。请修复不存在的节点引用、非法 output index、环形依赖，"
                    "并确保至少有一个终端输出节点，例如 SaveImage、PreviewImage、VHS_VideoCombine。"
                ),
            )

        if warnings:
            return DiagnoseItem(
                ok=True,
                severity="warning",
                evidence=evidence,
                suggestion="workflow 拓扑基本合法，但存在孤立/可疑节点。建议检查这些节点是否真的需要保留。",
            )

        return DiagnoseItem(
            ok=True,
            severity="info",
            evidence=evidence,
            suggestion="workflow JSON 拓扑合法。",
        )

    # ============================================================
    # 4. prompt_check
    # ============================================================

    def prompt_check(
        self,
        prompt_text: Optional[str],
        workflow: Optional[Dict[str, Any]],
    ) -> DiagnoseItem:
        evidence: Dict[str, Any] = {
            "prompt_text_present": bool(prompt_text and prompt_text.strip()),
            "prompt_text_length": len(prompt_text or ""),
            "workflow_text_inputs": [],
            "warnings": [],
        }

        if workflow is not None:
            try:
                api_prompt = self._normalize_workflow_to_api_prompt(workflow)
                for node_id, node in api_prompt.items():
                    inputs = node.get("inputs", {})
                    for key, value in inputs.items():
                        if key.lower() in {"text", "positive", "negative", "prompt"}:
                            if isinstance(value, str):
                                evidence["workflow_text_inputs"].append(
                                    {
                                        "node_id": node_id,
                                        "class_type": node.get("class_type"),
                                        "input": key,
                                        "length": len(value),
                                        "empty": not bool(value.strip()),
                                    }
                                )
            except Exception as e:
                evidence["warnings"].append(f"无法解析 workflow prompt 字段: {e}")

        empty_workflow_prompts = [
            x for x in evidence["workflow_text_inputs"] if x["empty"]
        ]

        if not evidence["prompt_text_present"] and not evidence["workflow_text_inputs"]:
            return DiagnoseItem(
                ok=True,
                severity="warning",
                evidence=evidence,
                suggestion="未发现外部 prompt_text，也未在 workflow 中发现明显文本输入。若是纯图生图/视频流程可忽略。",
            )

        if empty_workflow_prompts:
            return DiagnoseItem(
                ok=False,
                severity="warning",
                evidence={**evidence, "empty_workflow_prompts": empty_workflow_prompts},
                suggestion="发现空的文本输入节点。请补全 CLIPTextEncode 或相关节点的 text/positive/negative 字段。",
            )

        return DiagnoseItem(
            ok=True,
            severity="info",
            evidence=evidence,
            suggestion="prompt 文本检查通过。",
        )

    # ============================================================
    # 5. comfyui_check
    # ============================================================

    def comfyui_check(self) -> DiagnoseItem:
        evidence: Dict[str, Any] = {
            "url": self.comfyui_url,
            "reachable": False,
            "endpoint": None,
            "status_code": None,
            "latency_ms": None,
            "error": None,
        }

        if requests is None:
            return DiagnoseItem(
                ok=False,
                severity="critical",
                evidence={**evidence, "error": "requests is not installed"},
                suggestion="安装 requests：pip install requests。",
            )

        start = time.perf_counter()

        # ComfyUI 服务死了时，这里会触发 ConnectionError / Timeout
        endpoints = ["system_stats", "object_info"]
        last_error = None

        for ep in endpoints:
            try:
                resp = requests.get(
                    urljoin(self.comfyui_url, ep),
                    timeout=self.timeout,
                )
                latency_ms = round((time.perf_counter() - start) * 1000, 2)

                evidence.update(
                    {
                        "reachable": True,
                        "endpoint": ep,
                        "status_code": resp.status_code,
                        "latency_ms": latency_ms,
                    }
                )

                if 200 <= resp.status_code < 300:
                    return DiagnoseItem(
                        ok=True,
                        severity="info",
                        evidence=evidence,
                        suggestion="ComfyUI 服务在线。",
                    )

                last_error = f"{ep} returned HTTP {resp.status_code}"

            except requests.exceptions.ConnectionError as e:
                last_error = f"ConnectionError: {e}"
            except requests.exceptions.Timeout as e:
                last_error = f"Timeout after {self.timeout}s: {e}"
            except Exception as e:
                last_error = f"{type(e).__name__}: {e}"

        evidence["error"] = last_error

        host, port = self._parse_host_port(self.comfyui_url)
        evidence["tcp_probe"] = self._tcp_probe(host, port, timeout=self.timeout)

        return DiagnoseItem(
            ok=False,
            severity="critical",
            evidence=evidence,
            suggestion=(
                "ComfyUI 服务不可用。请确认 ComfyUI 已启动，地址和端口正确。"
                "默认启动方式：python main.py --listen 127.0.0.1 --port 8188。"
                "如果端口被占用，换端口后同步修改 comfyui_url。"
            ),
        )

    # ============================================================
    # 6. execution_check
    # ============================================================

    def execution_check(
        self,
        workflow: Optional[Dict[str, Any]],
        previous_results: Dict[str, DiagnoseItem],
    ) -> DiagnoseItem:
        evidence: Dict[str, Any] = {
            "execute_probe": self.execute_probe,
            "skipped": False,
            "reason": None,
            "prompt_id": None,
            "error": None,
        }

        if workflow is None:
            return DiagnoseItem(
                ok=False,
                severity="warning",
                evidence={**evidence, "skipped": True, "reason": "no workflow"},
                suggestion="未提供 workflow，跳过执行检查。",
            )

        if not previous_results.get("workflow_check", DiagnoseItem(False, "error", {}, "")).ok:
            return DiagnoseItem(
                ok=False,
                severity="error",
                evidence={**evidence, "skipped": True, "reason": "workflow invalid"},
                suggestion="workflow 拓扑非法，先修复 workflow_check 中的问题，再做执行检查。",
            )

        if not previous_results.get("comfyui_check", DiagnoseItem(False, "critical", {}, "")).ok:
            return DiagnoseItem(
                ok=False,
                severity="critical",
                evidence={**evidence, "skipped": True, "reason": "comfyui offline"},
                suggestion="ComfyUI 服务不可用，无法提交执行。请先启动 ComfyUI。",
            )

        if not self.execute_probe:
            return DiagnoseItem(
                ok=True,
                severity="warning",
                evidence={**evidence, "skipped": True, "reason": "execute_probe disabled"},
                suggestion=(
                    "已跳过真实执行探针，避免误触发生成任务。"
                    "如需提交到 ComfyUI 验证执行链路，请初始化时设置 execute_probe=True。"
                ),
            )

        try:
            api_prompt = self._normalize_workflow_to_api_prompt(workflow)
            client_id = str(uuid.uuid4())

            payload = {
                "prompt": api_prompt,
                "client_id": client_id,
            }

            resp = requests.post(
                urljoin(self.comfyui_url, "prompt"),
                json=payload,
                timeout=self.timeout,
            )

            evidence["status_code"] = resp.status_code
            evidence["response_text"] = resp.text[:2000]

            if not (200 <= resp.status_code < 300):
                return DiagnoseItem(
                    ok=False,
                    severity="error",
                    evidence=evidence,
                    suggestion="ComfyUI 拒绝执行请求。请查看 response_text，通常是节点参数、模型名或自定义节点缺失。",
                )

            data = resp.json()
            prompt_id = data.get("prompt_id")
            evidence["prompt_id"] = prompt_id

            if not prompt_id:
                return DiagnoseItem(
                    ok=False,
                    severity="error",
                    evidence=evidence,
                    suggestion="ComfyUI 未返回 prompt_id。请检查 /prompt 接口返回内容。",
                )

            return DiagnoseItem(
                ok=True,
                severity="info",
                evidence=evidence,
                suggestion="workflow 已成功提交到 ComfyUI 队列。",
            )

        except Exception as e:
            evidence["error"] = f"{type(e).__name__}: {e}"
            evidence["traceback"] = traceback.format_exc(limit=5)

            return DiagnoseItem(
                ok=False,
                severity="error",
                evidence=evidence,
                suggestion="执行探针失败。请重点检查 workflow 参数、自定义节点、模型文件和 ComfyUI 控制台错误。",
            )

    # ============================================================
    # Helpers
    # ============================================================

    def _load_workflow(self, workflow: Dict[str, Any] | str | Path) -> Dict[str, Any]:
        if isinstance(workflow, dict):
            return workflow

        path = Path(workflow).expanduser().resolve()
        with path.open("r", encoding="utf-8") as f:
            return json.load(f)

    def _normalize_workflow_to_api_prompt(self, workflow: Dict[str, Any]) -> Dict[str, Any]:
        """
        支持两种格式：
        1. ComfyUI API prompt:
           {
             "3": {"class_type": "KSampler", "inputs": {...}},
             "4": {"class_type": "CheckpointLoaderSimple", "inputs": {...}}
           }

        2. ComfyUI UI workflow:
           {
             "nodes": [...],
             "links": [...]
           }

        UI workflow 转 API prompt 只能做基础转换。
        更推荐调用 ComfyUI 自带 Save(API Format) 导出 API prompt。
        """

        # API prompt 格式
        if all(
            isinstance(k, str) and isinstance(v, dict) and "class_type" in v
            for k, v in workflow.items()
        ):
            return workflow

        # 有些包装格式是 {"prompt": {...}}
        if "prompt" in workflow and isinstance(workflow["prompt"], dict):
            return self._normalize_workflow_to_api_prompt(workflow["prompt"])

        # UI workflow 格式
        if "nodes" in workflow and isinstance(workflow["nodes"], list):
            return self._ui_workflow_to_api_prompt(workflow)

        raise ValueError("unknown workflow format")

    def _ui_workflow_to_api_prompt(self, workflow: Dict[str, Any]) -> Dict[str, Any]:
        nodes = workflow.get("nodes", [])
        links = workflow.get("links", [])

        link_map: Dict[int, Tuple[str, int]] = {}
        for link in links:
            # ComfyUI UI links 常见格式:
            # [link_id, origin_id, origin_slot, target_id, target_slot, type]
            if isinstance(link, list) and len(link) >= 5:
                link_id = int(link[0])
                origin_id = str(link[1])
                origin_slot = int(link[2])
                link_map[link_id] = (origin_id, origin_slot)

        api_prompt: Dict[str, Any] = {}

        for node in nodes:
            node_id = str(node.get("id"))
            class_type = node.get("type") or node.get("class_type")
            inputs: Dict[str, Any] = {}

            # widgets_values 无法可靠映射 input 名，这里只做拓扑转换。
            # 真实生产建议保存 API Format。
            for input_def in node.get("inputs", []) or []:
                name = input_def.get("name")
                link_id = input_def.get("link")
                if name and link_id is not None and link_id in link_map:
                    src_id, src_slot = link_map[link_id]
                    inputs[name] = [src_id, src_slot]

            api_prompt[node_id] = {
                "class_type": class_type,
                "inputs": inputs,
            }

        return api_prompt

    def _extract_required_models(self, workflow: Dict[str, Any]) -> List[Dict[str, Any]]:
        api_prompt = self._normalize_workflow_to_api_prompt(workflow)
        required: List[Dict[str, Any]] = []
        seen: Set[Tuple[str, str]] = set()

        for node_id, node in api_prompt.items():
            inputs = node.get("inputs", {})
            class_type = node.get("class_type")

            for key, value in inputs.items():
                if key not in self.MODEL_INPUT_KEYS:
                    continue

                if not isinstance(value, str) or not value.strip():
                    continue

                model_type = self.MODEL_INPUT_KEYS[key]
                filename = value.strip()
                dedupe_key = (model_type, filename)

                if dedupe_key in seen:
                    continue

                seen.add(dedupe_key)
                required.append(
                    {
                        "node_id": node_id,
                        "class_type": class_type,
                        "input": key,
                        "model_type": model_type,
                        "filename": filename,
                    }
                )

        return required

    def _find_model_file(self, model_type: str, filename: str) -> Optional[Path]:
        candidate = Path(filename)

        if candidate.is_absolute() and candidate.exists():
            return candidate.resolve()

        for rel_dir in self.model_dirs.get(model_type, []):
            base = (self.comfyui_root / rel_dir).resolve()

            direct = base / filename
            if direct.exists():
                return direct.resolve()

            # 兼容子目录，例如 "SDXL/foo.safetensors"
            nested = list(base.rglob(Path(filename).name)) if base.exists() else []
            for p in nested:
                if p.name == Path(filename).name:
                    return p.resolve()

        return None

    def _is_link(self, value: Any) -> bool:
        return (
            isinstance(value, list)
            and len(value) == 2
            and isinstance(value[0], (str, int))
        )

    def _detect_cycle(self, graph: Dict[str, Set[str]]) -> Optional[List[str]]:
        visiting: Set[str] = set()
        visited: Set[str] = set()
        stack: List[str] = []

        def dfs(node: str) -> Optional[List[str]]:
            visiting.add(node)
            stack.append(node)

            for nxt in graph.get(node, set()):
                if nxt in visited:
                    continue
                if nxt in visiting:
                    idx = stack.index(nxt)
                    return stack[idx:] + [nxt]
                result = dfs(nxt)
                if result:
                    return result

            visiting.remove(node)
            visited.add(node)
            stack.pop()
            return None

        for node in graph:
            if node not in visited:
                result = dfs(node)
                if result:
                    return result

        return None

    def _looks_like_source_node(self, node: Dict[str, Any]) -> bool:
        class_type = str(node.get("class_type", "")).lower()
        source_keywords = [
            "loader",
            "loadimage",
            "emptylatent",
            "primitive",
            "string",
            "integer",
            "float",
            "boolean",
            "cliptextencode",
            "note",
        ]
        return any(k in class_type for k in source_keywords)

    def _is_writable_dir(self, path: Path) -> bool:
        try:
            path.mkdir(parents=True, exist_ok=True)
            test_file = path / f".write_test_{uuid.uuid4().hex}"
            test_file.write_text("ok", encoding="utf-8")
            test_file.unlink(missing_ok=True)
            return True
        except Exception:
            return False

    def _parse_host_port(self, url: str) -> Tuple[str, int]:
        stripped = url.replace("http://", "").replace("https://", "").split("/")[0]
        if ":" in stripped:
            host, port = stripped.rsplit(":", 1)
            return host, int(port)
        return stripped, 443 if url.startswith("https://") else 80

    def _tcp_probe(self, host: str, port: int, timeout: float) -> Dict[str, Any]:
        result = {
            "host": host,
            "port": port,
            "open": False,
            "error": None,
        }

        try:
            with socket.create_connection((host, port), timeout=timeout):
                result["open"] = True
        except Exception as e:
            result["error"] = f"{type(e).__name__}: {e}"

        return result


class ComfyUIAgent:
    def __init__(
        self,
        comfyui_root: str | Path,
        comfyui_url: str = "http://127.0.0.1:8188",
    ) -> None:
        self.comfyui_root = Path(comfyui_root)
        self.comfyui_url = comfyui_url

    def diagnose(
        self,
        workflow: Optional[Dict[str, Any] | str | Path] = None,
        prompt_text: Optional[str] = None,
        execute_probe: bool = False,
    ) -> Dict[str, Dict[str, Any]]:
        diagnoser = ComfyUIDiagnoser(
            comfyui_root=self.comfyui_root,
            comfyui_url=self.comfyui_url,
            execute_probe=execute_probe,
        )

        return diagnoser.diagnose(
            workflow=workflow,
            prompt_text=prompt_text,
        )
