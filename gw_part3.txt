@dataclass
class ModelLoadPlan:
    mode: str  # "checkpoint" | "modular"
    unet: str | None = None
    clip: str | None = None
    clip2: str | None = None
    vae: str | None = None
    checkpoint: str | None = None


def choose_model_load_plan(task, env):
    if task.family in ["flux", "ltx", "wan"]:
        return ModelLoadPlan(
            mode="modular",
            unet=env.pick("unet"),
            clip=env.pick("clip"),
            clip2=env.pick_optional("clip"),
            vae=env.pick("vae"),
        )

    return ModelLoadPlan(
        mode="checkpoint",
        checkpoint=env.pick("checkpoints"),
