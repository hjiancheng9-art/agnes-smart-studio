"""Connect to Edge via CDP and read ChatGPT conversation."""
import sys, os, time, json
sys.stdout.reconfigure(encoding='utf-8', errors='replace')

from playwright.sync_api import sync_playwright

CHATGPT_URL = "https://chatgpt.com/c/6a49f474-f5f8-83e8-b9ae-fe5e13f8b1aa"
OUTPUT_DIR = r"C:\Users\huangjiancheng\agnes-smart-studio\output"

with sync_playwright() as p:
    # Connect to existing Edge via CDP
    browser = p.chromium.connect_over_cdp("http://127.0.0.1:9222")
    
    # Get default context (Edge's existing profile)
    default_context = browser.contexts[0]
    
    # Open a new page in this context (inherits Edge's cookies)
    page = default_context.new_page()
    
    print("Navigating to ChatGPT conversation...")
    try:
        page.goto(CHATGPT_URL, wait_until='domcontentloaded', timeout=20000)
    except Exception as e:
        print(f"Navigation timeout or error: {e}")
    
    time.sleep(8)
    
    # Screenshot
    page.screenshot(path=os.path.join(OUTPUT_DIR, 'chatgpt_session.png'))
    title = page.title()
    url = page.url
    print(f"TITLE: {title}")
    print(f"URL: {url}")
    
    if 'auth' in url or 'login' in url:
        print("STATUS: NEED_LOGIN - Edge is not logged into ChatGPT")
    else:
        print("STATUS: LOGGED_IN")
        
        # Scroll to load all messages
        for _ in range(5):
            try:
                page.evaluate('window.scrollTo(0, document.body.scrollHeight)')
            except:
                pass
            time.sleep(1)
        
        # Extract messages
        messages = page.query_selector_all('[data-message-author-role]')
        if not messages:
            messages = page.query_selector_all('[data-testid="conversation-turn"]')
        
        print(f"MSG_COUNT: {len(messages)}")
        
        conversation = []
        for i, msg in enumerate(messages):
            role = msg.get_attribute('data-message-author-role') or 'unknown'
            content = msg.inner_text()
            conversation.append({'role': role, 'content': content})
            print(f"MSG_{i}|{role}|{len(content)}chars")
            print(content[:500])
            print("---END---")
        
        with open(os.path.join(OUTPUT_DIR, 'chatgpt_conversation.json'), 'w', encoding='utf-8') as f:
            json.dump(conversation, f, ensure_ascii=False, indent=2)
        
        print(f"DONE: {len(conversation)} messages saved")
    
    page.close()
    browser.close()
