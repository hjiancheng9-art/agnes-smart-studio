1. 先给 brain.py 加 LLM 适配器

compile() 需要 Callable[[str], str]，但 brain.chat() 返回 dict，所以先补这个。

# brain.py

from typing import Any, Callable, Optional


class Brain:
    # 你原来的 Brain 内容保留

    def chat(self, message: str, **kwargs) -> Any:
        """
        原有接口：可能返回 dict / str / message object
        """
        raise NotImplementedError

    def to_llm_chat_func(self, *, system_prompt: Optional[str] = None) -> Callable[[str], str]:
        """
        适配 FlowCompiler.compile(nl, llm_chat_func)
        返回一个只接收 prompt、只返回 str 的函数。
        """

        def _llm_chat(prompt: str) -> str:
            final_prompt = prompt
            if system_prompt:
                final_prompt = f"{system_prompt.strip()}\n\n{prompt.strip()}"

            result = self.chat(final_prompt)
            return self._extract_text(result)

        return _llm_chat

    @staticmethod
    def _extract_text(result: Any) -> str:
        if result is None:
            return ""

        if isinstance(result, str):
            return result

        if isinstance(result, dict):
            for key in (
                "content",
                "text",
                "message",
                "response",
                "answer",
                "output",
            ):
                value = result.get(key)
                if isinstance(value, str):
                    return value

            # OpenAI / DeepSeek / Claude 类格式兼容
            choices = result.get("choices")
            if isinstance(choices, list) and choices:
                first = choices[0]
                if isinstance(first, dict):
                    msg = first.get("message")
                    if isinstance(msg, dict) and isinstance(msg.get("content"), str):
                        return msg["content"]
                    if isinstance(first.get("text"), str):
                        return first["text"]

            # 兜底：避免 compile() 直接炸
            return str(result)

        if hasattr(result, "content"):
            return str(result.content)

        return str(result)

注意：/object_info 应该用于验证真实节点是否存在，但不要把完整 object_info 全塞进 prompt，不然 LLM 会被噪音淹没。

运行
# node_catalog.py

from __future__ import annotations

from dataclasses import dataclass
from typing import Any, Dict, List, Optional, Set
import time
import requests


DEFAULT_CORE_NODES = {
    "CheckpointLoaderSimple",
    "CLIPTextEncode",
    "KSampler",
    "EmptyLatentImage",
    "VAEDecode",
    "SaveImage",
    "LoadImage",
    "VAEEncode",
    "PreviewImage",
    "LoraLoader",
    "ControlNetLoader",
    "ControlNetApplyAdvanced",
}


@dataclass
class NodeSpec:
    class_type: str
    required_inputs: List[str]
    optional_inputs: List[str]
    outputs: List[str]


class NodeCatalog:
    def __init__(
        self,
        comfyui_url: str = "http://127.0.0.1:8188",
        cache_ttl_sec: int = 300,
        fallback_nodes: Optional[Set[str]] = None,
    ):
        self.comfyui_url = comfyui_url.rstrip("/")
        self.cache_ttl_sec = cache_ttl_sec
        self.fallback_nodes = fallback_nodes or DEFAULT_CORE_NODES

        self._nodes: Dict[str, NodeSpec] = {}
        self._last_refresh = 0.0
        self._last_error: Optional[str] = None

    @property
    def last_error(self) -> Optional[str]:
        return self._last_error

    def get_allowed_node_types(self) -> Set[str]:
        if self._nodes:
            return set(self._nodes.keys())
        return set(self.fallback_nodes)

    def refresh_from_comfyui(self, *, force: bool = False, timeout: float = 8.0) -> bool:
        now = time.time()
        if not force and self._nodes and now - self._last_refresh < self.cache_ttl_sec:
            return True

        try:
            resp = requests.get(f"{self.comfyui_url}/object_info", timeout=timeout)
            resp.raise_for_status()
            raw = resp.json()

            parsed: Dict[str, NodeSpec] = {}

            for class_type, meta in raw.items():
                inputs = meta.get("input", {}) if isinstance(meta, dict) else {}
                required = inputs.get("required", {}) or {}
                optional = inputs.get("optional", {}) or {}

                outputs = meta.get("output", []) if isinstance(meta, dict) else []
                if not isinstance(outputs, list):
                    outputs = [str(outputs)]

                parsed[class_type] = NodeSpec(
                    class_type=class_type,
                    required_inputs=list(required.keys()),
                    optional_inputs=list(optional.keys()),
                    outputs=[str(x) for x in outputs],
                )

            if parsed:
                self._nodes = parsed
                self._last_refresh = now
                self._last_error = None
                return True

            self._last_error = "ComfyUI /object_info returned empty node catalog"
            return False

        except Exception as e:
            self._last_error = f"Failed to refresh node catalog: {e}"
            return False

    def compact_prompt_catalog(self, *, limit: int = 80) -> str:
        """
        给 LLM 用的短节点目录。
        不要塞完整 object_info。
        """
        if not self._nodes:
            nodes = sorted(self.fallback_nodes)
            return "\n".join(f"- {name}" for name in nodes)

        preferred = []
        for name in DEFAULT_CORE_NODES:
            if name in self._nodes:
                preferred.append(name)

        remaining = sorted(name for name in self._nodes.keys() if name not in preferred)
        selected = (preferred + remaining)[:limit]

        lines = []
        for name in selected:
            spec = self._nodes[name]
            req = ", ".join(spec.required_inputs[:8])
            opt = ", ".join(spec.optional_inputs[:8])
            out = ", ".join(spec.outputs[:8])

            parts = [f"- {name}"]
            if req:
                parts.append(f"required: {req}")
            if opt:
                parts.append(f"optional: {opt}")
            if out:
                parts.append(f"outputs: {out}")

            lines.append(" | ".join(parts))

        return "\n".join(lines)

from typing import Callable, Dict, Any, Optional, Set


class WorkflowValidator:
    def __init__(self, allowed_node_types: Optional[Set[str]] = None):
        self.allowed_node_types = allowed_node_types or {
            "CheckpointLoaderSimple",
            "CLIPTextEncode",
            "KSampler",
            "EmptyLatentImage",
            "VAEDecode",
            "SaveImage",
            "LoadImage",
            "VAEEncode",
            "PreviewImage",
            "LoraLoader",
            "ControlNetLoader",
            "ControlNetApplyAdvanced",
        }

    def update_allowed_node_types(self, node_types: Set[str]) -> None:
        if node_types:
            self.allowed_node_types = set(node_types)

    def validate(self, workflow: Dict[str, Any]) -> Dict[str, Any]:
        """
        你原有的 JSON Schema 校验 + 白名单 + 连通性检查保留。
        这里只示例白名单部分。
        """
        errors = []

        if not isinstance(workflow, dict):
            return {
                "ok": False,
                "errors": ["workflow must be a dict"],
            }

        for node_id, node in workflow.items():
            if not isinstance(node, dict):
                errors.append(f"Node {node_id} must be object")
                continue

            class_type = node.get("class_type")
            if not class_type:
                errors.append(f"Node {node_id} missing class_type")
                continue

            if class_type not in self.allowed_node_types:
                errors.append(
                    f"Node {node_id} uses unsupported class_type: {class_type}"
                )

        if errors:
            return {
                "ok": False,
                "errors": errors,
            }

        # 接你原来的 schema + graph connectivity 检查
        return {
            "ok": True,
            "errors": [],
        }


class FlowCompiler:
    def __init__(self, validator: Optional[WorkflowValidator] = None):
        self.validator = validator or WorkflowValidator()

    def compile(
        self,
        nl: str,
        llm_chat_func: Callable[[str], str],
        *,
        node_catalog_text: Optional[str] = None,
    ) -> Dict[str, Any]:
        prompt = self._build_prompt(nl, node_catalog_text=node_catalog_text)

        raw = llm_chat_func(prompt)
        workflow = self._extract_json(raw)

        validation = self.validator.validate(workflow)

        return {
            "ok": validation.get("ok", False),
            "workflow": workflow if validation.get("ok") else None,
            "raw_workflow": workflow,
            "validation": validation,
            "llm_raw": raw,
        }

    def _build_prompt(self, nl: str, *, node_catalog_text: Optional[str]) -> str:
        catalog = node_catalog_text or "Use only common ComfyUI core nodes."

        return f"""
你是 ComfyUI Workflow JSON 编译器。

任务：
把用户自然语言需求编译成 ComfyUI API workflow JSON。

硬性规则：
1. 只输出 JSON，不要 Markdown，不要解释。
2. JSON 顶层必须是对象，key 是节点 id 字符串。
3. 每个节点必须包含：
   - class_type
   - inputs
4. class_type 必须来自允许节点目录。
5. 节点之间的连接使用 ComfyUI API 格式：
   "input_name": ["node_id", output_index]
6. 不要编造节点名。
7. 不确定时优先使用最基础 txt2img 工作流：
   CheckpointLoaderSimple → CLIPTextEncode → EmptyLatentImage → KSampler → VAEDecode → SaveImage

允许节点目录：
{catalog}

用户需求：
{nl}

只输出 workflow JSON：
""".strip()

    def _extract_json(self, text: str) -> Dict[str, Any]:
        """
        你已有实现保留。
        """
        raise NotImplementedError

这是最关键的一步。之后所有地方都调这个。

# agent.py

from typing import Any, Dict, Optional

from flow_compiler import FlowCompiler, WorkflowValidator
from node_catalog import NodeCatalog


class ComfyUIAgent:
    def __init__(
        self,
        brain=None,
        comfyui_url: str = "http://127.0.0.1:8188",
        flow_compiler: Optional[FlowCompiler] = None,
        node_catalog: Optional[NodeCatalog] = None,
    ):
        self.brain = brain
        self.comfyui_url = comfyui_url.rstrip("/")

        self.node_catalog = node_catalog or NodeCatalog(self.comfyui_url)
        self.flow_validator = WorkflowValidator()
        self.flow_compiler = flow_compiler or FlowCompiler(self.flow_validator)

    def validate(self, workflow: Dict[str, Any]) -> Dict[str, Any]:
        self._refresh_node_catalog_best_effort()
        return self.flow_validator.validate(workflow)

    def compile_workflow(
        self,
        nl: str,
        *,
        brain=None,
        refresh_nodes: bool = True,
    ) -> Dict[str, Any]:
        """
        NL → ComfyUI workflow 的唯一入口。
        server.py / showrunner / agent_flow 都只调用这个。
        """
        if not nl or not nl.strip():
            return {
                "ok": False,
                "error": "nl is required",
            }

        active_brain = brain or self.brain
        if active_brain is None:
            return {
                "ok": False,
                "error": "No brain configured for workflow compilation",
            }

        if refresh_nodes:
            self._refresh_node_catalog_best_effort()

        if not hasattr(active_brain, "to_llm_chat_func"):
            return {
                "ok": False,
                "error": "Brain does not implement to_llm_chat_func()",
            }

        llm_chat_func = active_brain.to_llm_chat_func(
            system_prompt=(
                "你是严格的 ComfyUI workflow JSON 编译器。"
                "只输出合法 JSON，不输出解释。"
            )
        )

        node_catalog_text = self.node_catalog.compact_prompt_catalog()

        result = self.flow_compiler.compile(
            nl,
            llm_chat_func,
            node_catalog_text=node_catalog_text,
        )

        return {
            "ok": bool(result.get("ok")),
            "workflow": result.get("workflow"),
            "raw_workflow": result.get("raw_workflow"),
            "validation": result.get("validation"),
            "node_catalog_error": self.node_catalog.last_error,
            "llm_raw": result.get("llm_raw"),
        }

    def _refresh_node_catalog_best_effort(self) -> None:
        self.node_catalog.refresh_from_comfyui(force=False)

        allowed = self.node_catalog.get_allowed_node_types()
        self.flow_validator.update_allowed_node_types(allowed)

# server.py

from flask import Flask, request, jsonify

app = Flask(__name__)


def create_server(agent):
    @app.post("/compile")
    def compile_workflow():
        payload = request.get_json(silent=True) or {}

        nl = payload.get("nl") or payload.get("message") or payload.get("prompt")
        refresh_nodes = bool(payload.get("refresh_nodes", True))

        if not nl:
            return jsonify({
                "ok": False,
                "error": "Missing required field: nl",
            }), 400

        result = agent.compile_workflow(
            nl,
            refresh_nodes=refresh_nodes,
        )

        status = 200 if result.get("ok") else 422
        return jsonify(result), status

    @app.get("/nodes")
    def list_nodes():
        agent.node_catalog.refresh_from_comfyui(force=True)
        nodes = sorted(agent.node_catalog.get_allowed_node_types())

        return jsonify({
            "ok": True,
            "count": len(nodes),
            "nodes": nodes,
            "error": agent.node_catalog.last_error,
        })

    return app

from typing import Any, Dict


class ShowrunnerOrchestrator:
    def __init__(self, comfyui_agent):
        self.comfyui_agent = comfyui_agent

    def _produce_shot(self, shot: Dict[str, Any]) -> Dict[str, Any]:
        """
        shot 示例：
        {
            "id": "shot_001",
            "visual_prompt": "...",
            "duration": 4,
            "camera": "...",
            "style": "..."
        }
        """

        nl = self._shot_to_workflow_prompt(shot)

        compile_result = self.comfyui_agent.compile_workflow(nl)

        if not compile_result.get("ok"):
            return {
                "ok": False,
                "stage": "compile_workflow",
                "shot": shot,
                "error": compile_result.get("error"),
                "validation": compile_result.get("validation"),
                "llm_raw": compile_result.get("llm_raw"),
            }

        workflow = compile_result["workflow"]

        # 如果你的 agent.execute() 已经存在，就接这里
        execute_result = self.comfyui_agent.execute(workflow)

        return {
            "ok": bool(execute_result.get("ok", True)),
            "stage": "produce_shot",
            "shot": shot,
            "workflow": workflow,
            "compile": compile_result,
            "execute": execute_result,
        }

    def _shot_to_workflow_prompt(self, shot: Dict[str, Any]) -> str:
        visual_prompt = shot.get("visual_prompt") or shot.get("prompt") or ""
        negative_prompt = shot.get("negative_prompt") or (
            "low quality, blurry, bad anatomy, deformed, watermark, text"
        )
        width = shot.get("width", 768)
        height = shot.get("height", 1344)
        steps = shot.get("steps", 24)
        cfg = shot.get("cfg", 7)

        return f"""
请生成一个 ComfyUI txt2img workflow。

画面提示词：
{visual_prompt}

负面提示词：
{negative_prompt}

参数：
- width: {width}
- height: {height}
- steps: {steps}
- cfg: {cfg}
- sampler: euler 或 dpmpp_2m
- scheduler: normal 或 karras

要求：
- 使用基础 txt2img 工作流
- 必须包含 SaveImage
- 输出 ComfyUI API workflow JSON
""".strip()
