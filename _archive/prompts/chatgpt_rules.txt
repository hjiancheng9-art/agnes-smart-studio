六、规则生命周期：定义 → 注册 → 匹配 → 执行 → 审计
1. 规则定义

JavaScript{  id: "core.never_fake_success",  title: "Never fake success",  description: "A stage cannot be marked completed unless real outputs and evidence exist.",  scope: ["stage", "task", "artifact"],  severity: "block",  priority: 1000,  appliesTo: {    stages: ["*"],    modes: ["production", "autoflow"]  },  when: "beforeStageComplete",  check(ctx) {    return ctx.stage.status === "completed"      && ctx.outputs.realArtifactCount > 0      && ctx.evidence.exists === true      && ctx.outputs.containsPlaceholder === false;  },  onFail: {    action: "block",    message: "Cannot mark stage complete without real artifact and evidence."  }}

2. 规则注册
rule-registry.js：
JavaScriptclass RuleRegistry {  constructor() {    this.rules = new Map();  }  register(rule) {    validateRule(rule);    if (this.rules.has(rule.id)) {      throw new Error(`Duplicate rule id: ${rule.id}`);    }    this.rules.set(rule.id, rule);  }  getRulesFor(eventName, ctx) {    return [...this.rules.values()]      .filter(rule => rule.when === eventName)      .filter(rule => matchesScope(rule, ctx))      .sort((a, b) => b.priority - a.priority);  }}
加载：
JavaScriptregistry.registerMany(coreHardRules);registry.registerMany(stageTransitionRules);registry.registerMany(artifactRules);registry.registerMany(cruxRules);registry.registerMany(deliveryRules);

"allow" : rule.onFail.action,      message: passed ? "" : rule.onFail.message    };    results.push(result);    await ruleAudit.write(ctx, result);    if (!passed && rule.onFail.action === "block") {      throw new RuleBlockedError(result);    }  }  return results;}

十、CRUX 总控如何接入规则层
1. CRUX 不能绕过规则层
CRUX 任务调度必须经过这些规则点：
beforeCruxDispatchafterCruxResultbeforeArtifactCommitbeforeStageComplete

JavaScript{  taskId: "task_001",  runId: "run_001",  stageId: "9-animation",  mode: "production",  ruleSnapshotId: "rs_001",  knowledgeSnapshotId: "ks_001",  requiredRules: [    "core.never_fake_success",    "artifact.no_placeholder_output",    "crux.must_return_evidence"  ],  constraints: {    allowMock: false,    requireEvidence: true,    requireArtifactHash: true  }}
