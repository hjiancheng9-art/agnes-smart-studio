Python
运行
async def execute(self, request: str, context: dict | None = None, policy=None) -> dict:
    context = context or {}

    if policy is None:
        policy = await self.policy_router.route(request, context)

    goal = await self.toolbus.call("create_goal", {
        "request": request,
        "policy": policy.to_dict() if hasattr(policy, "to_dict") else policy,
    })

    if self.config.capability_runtime.enabled:
        runtime_decision = self.capability_router.route(
            request=request,
            goal=goal,
            policy=policy,
            context=context,
        )

        if self._runtime_enabled(runtime_decision.runtime_name):
            try:
                return await self._execute_with_runtime(
                    request=request,
                    context=context,
                    policy=policy,
                    goal=goal,
                    runtime_decision=runtime_decision,
                )
            except Exception as exc:
                self.trace_store.record_error(
                    phase="capability_runtime",
                    error=repr(exc),
                    fallback="legacy_deliberate_workflow",
                )

    return await self._execute_legacy(
        request=request,
        context=context,
        policy=policy,
        goal=goal,
    )

重点：

Runtime 失败必须 fallback 到 legacy DeliberateWorkflow。
不要让 Runtime 失败直接炸掉主聊天。
